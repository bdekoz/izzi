75 /// Paint the edges of a physical page.
76 void
77 paint_edges_with_char_index(svg_element& obj, const area<> a,
78  const char firstc)
79 {
80  // Paint edges.
81  const std::locale loc("");
82  const bool alphap = std::isalpha(firstc, loc);
83 
84  // Position the first character from a to z to x values 0 to 25
85  uint cidx(0);
86  if (alphap)
87  cidx = (static_cast<uint>(firstc) % 65) - 32 + 1;
88  auto deltax = (a._M_width / 27) * cidx;
89  auto deltay = (a._M_height / 27) * cidx;
90 
91  style estyl = svg::k::b_style;
92  estyl.set_colors(color::duboisgreen2);
93 
94  // NB: 0.125 is a common bleed == 12 pixels. Plus 2 pixel "on page" -> 14.
95  constexpr uint len = 14;
96 
97  // Top edge, right to left starting outer to inner
98  point_2d_to_rect(obj, a._M_width - deltax - len - 10, 0, estyl, 10, len);
99 
100  // Bottom edge, left to right starting innner to outer.
101  point_2d_to_rect(obj, deltax + len, a._M_height - len, estyl, 10, len);
102 
103  // Right side edge, up from bottom to top.
104  point_2d_to_rect(obj, a._M_width - len, a._M_height - deltay - len - 10,
105  estyl, len, 10);
106 }

109 /// Import svg file, convert it to svg_element for insertion.
110 /// ifile is a plain SVG file with a 1:1 aspect ratio.
111 string
112 file_to_svg_insert(const string ifile)
113 {
114  string isvg;
115 
116  // Read SVG to insert.
117  std::ifstream ifs(ifile);
118  if (ifs.good())
119  {
120  // Strip out any XML version line in the SVG file.
121  // Search for and discard lines with "xml version", iff exists
122  string xmlheader;
123  getline(ifs, xmlheader);
124  if (xmlheader.find("xml version") == string::npos)
125  ifs.seekg(0, ifs.beg);
126 
127  std::ostringstream oss;
128  oss << ifs.rdbuf();
129  isvg = oss.str();
130  }
131  else
132  {
133  string m("svg_file_to_svg_insert:: insert nested SVG failed ");
134  m += ifile;
135  m += k::newline;
136  throw std::runtime_error(m);
137  }
138  return isvg;
139 }

170 /// Embed svg in group element.
171 /// @origin is where glyph placement is inside containing svg element.
172 /// @origsize is original file width/height constant
173 /// @isize is final width/height
174 /// @isvg is the string from one of the two functions above (*_to_svg_insert).
175 /// @syl is overide style information: defaults to no_style.
176 ///
177 /// NB This only works is the file has no styles set in svg, group, or
178 /// individual element definitions (like circle, path, rectangle,
179 /// etc.).
180 ///
181 /// See: https://developer.mozilla.org/en-US/docs/Web/SVG/Element/svg
182 svg_element
183 insert_svg_at(svg_element& obj, const string isvg,
184  const point_2t origin, const double origsize, const double isize,
185  const double angled = 0, const style& styl = k::no_style)
186 {
187  // offset
188  auto [ objx, objy ] = origin;
189  const int x = objx - (isize / 2);
190  const int y = objy - (isize / 2);
191  string xformtranslate(transform::translate(x, y));
192 
193  // scaled
194  string xformscale;
195  if (isize != origsize)
196  {
197  const double scalex(isize / origsize);
198  xformscale = transform::scale(scalex);
199  }
200 
201  // rotated
202  string xformrotate;
203  if (angled != 0)
204  {
205  xformrotate = transform::rotate(90 - angled, objx, objy);
206  }
207 
208  // Order of transformations matters...
209  string ts(xformrotate + k::space + xformtranslate + k::space + xformscale);
210 
211  group_element gsvg;
212  gsvg.start_element("inset svg", transform(), ts, styl);
213  gsvg.add_raw(isvg);
214  gsvg.finish_element();
215  obj.add_element(gsvg);
216 
217  return obj;
218 }

221 /// Composite frame on bleed.
222 /// For printed objects with a center gutter, some intra-page
223 /// adjustments are necessary.
224 /// @slxt is odd/even (left/right)
225 /// @bleedin is bleed size for one edge in inches.
226 void
227 composite_bleed_areas(svg_element& obj,
228  const svg::k::select slxt, const double bleedin)
229 {
230  const double bleedpx = get_dpi() * bleedin;
231  if (slxt == svg::k::select::odd)
232  {
233  // Individual glyph shapes grid.
234  // Left, center vertical and move to right edge horizontal.
235  const point_2t p = { bleedpx, bleedpx };
236  obj.start_element(p, obj._M_area);
237  }
238  if (slxt == svg::k::select::even)
239  {
240  // Radial metadata dimensions grid.
241  // Right, center vertical and move to left edge horizontal
242  const point_2t p = { 0, bleedpx };
243  obj.start_element(p, obj._M_area);
244  }
245 }
