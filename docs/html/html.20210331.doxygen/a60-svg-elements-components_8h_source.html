44 svg_element::start_element()
45 {
46  const string start = R"_delimiter_(<?xml version="1.0" encoding="utf-8"?>
47 <!DOCTYPE svg PUBLIC "-//W3C//DTD SVG 1.1//EN" "http://www.w3.org/Graphics/SVG/1.1/DTD/svg11.dtd">
48 <svg version="1.1"
49  id="svg2" xml:space="preserve"
50  xmlns:dc="http://purl.org/dc/elements/1.1/"
51  xmlns:cc="http://creativecommons.org/ns#"
52  xmlns:svg="http://www.w3.org/2000/svg"
53  xmlns="http://www.w3.org/2000/svg"
54  xmlns:xlink="http://www.w3.org/1999/xlink"
55 )_delimiter_";
56 
57  const string unit("__unit");
58  const string width("__width");
59  const string height("__height");
60 
61  string strip = R"_delimiter_(x="0__unit" y="0__unit"
62 width="__width__unit" height="__height__unit"
63 viewBox="0 0 __width __height" enable-background="new 0 0 __width __height">
64 )_delimiter_";
65 
66  string_replace(strip, unit, to_string(_M_unit));
67  string_replace(strip, width, std::to_string(_M_area._M_width));
68  string_replace(strip, height, std::to_string(_M_area._M_height));
69 
70  _M_sstream << start;
71  _M_sstream << strip << std::endl;
72 }
