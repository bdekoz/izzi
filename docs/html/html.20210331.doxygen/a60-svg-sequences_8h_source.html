32 /**
33  Fade rectangle fill from transparent to color, of duration sec, given fps
34  starts on white (klr), fades to klr color in sec
35  can use with image behind
36 */
37 strings
38 fade_to_color_seq(const rect_element::data& dr, const color klr,
39  size_t fps = 30, double sec = 1.0, double maxopac = 1.0)
40 {
41  // Calc number of frames needed.
42  size_t framesn = fps * sec;
43  double step = maxopac / framesn;
44 
45  // Start rectangle as transparent, fade to color.
46  strings ret;
47  style sty = { klr, 0.0, klr, 0.0, 0 };
48  for (size_t i = 0; i < framesn; ++i)
49  {
50  // Foreground rectangle.
51  rect_element rsvg;
52 
53  double opacity = i * step;
54  sty._M_fill_opacity = opacity;
55 
56  rsvg.start_element();
57  rsvg.add_data(dr);
58  rsvg.add_style(sty);
59  rsvg.finish_element();
60  ret.push_back(rsvg.str());
61  }
62 
63  return ret;
64 }

153 /// wink sequence of duration sec, given fps to color klr
154 /// starts on background transparent, so can use with an image behind.
155 /// at duration twhen
156 /// winks specific rectangle (r) this much (maxclose) for seconds (winksec)
157 strings
158 wink_to_color_seq(const rect_element::data& dr, const color klr,
159  size_t fps, double sec,
160  double twhen, double maxclose, double winksec)
161 {
162  const int rows = dr._M_height;
163 
164  // Calc number of frames needed.
165  size_t framesn = fps * sec;
166  size_t startn = fps * twhen;
167  size_t winkframesn = (fps * winksec) + 1;
168  size_t winkframeshalfn = winkframesn / 2;
169 
170  // End with backgf.
171  if (startn + winkframesn >= framesn)
172  throw std::logic_error("wink:: desired frames exceed total duration");
173 
174  strings ret;
175  double opaque(1.0);
176  double transparent(0.0);
177  style sty = { klr, 0.0, klr, 0.0, 0 };
178 
179  // Generate intro.
180  for (size_t i = 0; i < startn; ++i)
181  {
182  sty._M_fill_opacity = transparent;
183  rect_element rsvg;
184  rsvg.start_element();
185  rsvg.add_data(dr);
186  rsvg.add_style(sty);
187  rsvg.finish_element();
188  ret.push_back(rsvg.str());
189  }
190 
191  // Make two "eyelid" rectangles, one upper and the other lower.
192  double ymid = dr._M_height / 2;
193  double ymax = ymid * maxclose;
194 
195  // grow down
196  rect_element::data dhi = { dr._M_x_origin, dr._M_y_origin, dr._M_width, 0 };
197 
198  // grow up
199  rect_element::data dlo = { dr._M_x_origin, dr._M_y_origin + rows,
200  dr._M_width, 0 };
201 
202  // Wink.
203  // Step is ammount to move from eyes open to closed and back again.
204  const double step = 2 * ymax / winkframesn;
205 
206  // Wink down.
207  for (size_t j = 0; j < winkframeshalfn; ++j)
208  {
209  sty._M_fill_opacity = opaque;
210  double offsetn = j * step;
211 
212  // Double rainbow, y'all.
213  rect_element r1;
214  r1.start_element();
215  dhi._M_height = offsetn;
216  r1.add_data(dhi);
217  r1.add_style(sty);
218  r1.finish_element();
219 
220  rect_element r2;
221  r2.start_element();
222  dlo._M_y_origin = dr._M_y_origin + rows - offsetn;
223  dlo._M_height = offsetn;
224  r2.add_data(dlo);
225  r2.add_style(sty);
226  r2.finish_element();
227 
228  ret.push_back(r1.str() + r2.str());
229  }
230 
231  // Wink up.
232  for (size_t j = 0; j < winkframeshalfn; ++j)
233  {
234  sty._M_fill_opacity = opaque;
235  double offsetn = (winkframeshalfn - j) * step;
236 
237  // Double rainbow, y'all.
238  rect_element r1;
239  r1.start_element();
240  dhi._M_height = offsetn;
241  r1.add_data(dhi);
242  r1.add_style(sty);
243  r1.finish_element();
244 
245  rect_element r2;
246  r2.start_element();
247  dlo._M_y_origin = dr._M_y_origin + rows - offsetn;
248  dlo._M_height = offsetn;
249  r2.add_data(dlo);
250  r2.add_style(sty);
251  r2.finish_element();
252 
253  ret.push_back(r1.str() + r2.str());
254  }
255 
256  // Generate outtro.
257  for (size_t i = startn + winkframesn; i < framesn; ++i)
258  {
259  sty._M_fill_opacity = transparent;
260  rect_element rsvg;
261  rsvg.start_element();
262  rsvg.add_data(dr);
263  rsvg.add_style(sty);
264  rsvg.finish_element();
265  ret.push_back(rsvg.str());
266  }
267 
268  return ret;
269 }

272 /// Simulated vertical roll, with fades.
273 /// r == frame size
274 strings
275 vertical_sync_roll_seq(const rect_element::data& drin, const color klr,
276  size_t /*fps = 30*/, double step = 10,
277  int blursz = 200, int solidsz = 133, double opac = 0.6)
278 {
279  // Composition of two rectangular elements:
280  // (background) 60 pixel wide rectangular blur 10 pixels, 60% opacity
281  // (foreground) 20 pixels wide centered solid color overlay
282  const int blurr = 20; // blur radius
283  // const int blursz = 200; // 60, 100, 160, 200
284  // const int solidsz = 133; // 20, 40, 80, 133
285  const double vblursz = blursz / 2;
286  const double vsolidsz = solidsz / 2;
287 
288  // One full sweep is half-bottom to offscreen top.
289  double x = (drin._M_height + (2 * blursz) + (3 * blurr)) / step;
290  size_t framesn = static_cast<size_t>(x);
291 
292  style sty = { klr, 1.0, klr, 0.0, 0 };
293 
294  // rect_elementangles start at bottom off-screen and sweep up.
295  // SVG rect area is rect 00 top left
296  // SVG rect is drawn down and left from starting position.
297 
298  // Blur rectangles. The SVG gaussian blur gives a hard edge on the
299  // top corner. So, to have an even fade, double it and center under
300  // the solid rectangle.
301  rect_element::data drblurb(drin);
302  drblurb._M_y_origin = drin._M_y_origin + drin._M_height;
303  drblurb._M_height = blursz;
304 
305  rect_element::data drblurt(drin);
306  drblurt._M_y_origin = drin._M_y_origin + drin._M_height - blursz;
307  drblurt._M_height = blursz;
308 
309  // Solid rectangle.
310  rect_element::data drs(drin);
311  drs._M_y_origin = drin._M_y_origin + drin._M_height - vsolidsz;
312  drs._M_height = solidsz;
313 
314  strings ret;
315  for (size_t i = 0; i < framesn; ++i)
316  {
317  double stepn = i * step;
318 
319  // Dual blur.
320  sty._M_fill_opacity = opac;
321 
322  rect_element rsvg1;
323  size_type yblurb = static_cast<int>(drblurb._M_y_origin - stepn);
324  rsvg1.start_element();
325  rsvg1.add_data(drblurb);
326  rsvg1.add_filter("20y");
327  rsvg1.add_style(sty);
328  rsvg1.finish_element();
329 
330  rect_element rsvg3;
331  // size_type yblurt = static_cast<int>(drblurt._M_y_origin - stepn);
332  rsvg3.start_element();
333  rsvg3.add_data(drblurt);
334  rsvg3.add_filter("20y");
335 
336  std::ostringstream ostrt;
337  ostrt << "rotate(180, 960, " << yblurb - vblursz << ")";
338  rsvg3.add_transform(ostrt.str());
339  rsvg3.add_style(sty);
340  rsvg3.finish_element();
341 
342  string scene(rsvg1.str() + rsvg3.str());
343 
344  // Foreground rectangle.
345  sty._M_fill_opacity = opac + 0.20;
346  rect_element rsvgfg;
347  // size_type y = static_cast<int>(drs._M_y_origin - stepn);
348  rsvgfg.start_element();
349  rsvgfg.add_data(drs);
350  rsvgfg.add_style(sty);
351  rsvgfg.finish_element();
352 
353  ret.push_back(scene + rsvgfg.str());
354  }
355 
356  return ret;
357 }

360 /// Randomly create grid of [maxwidth] x [maxheight] grid, and fill with dots.
361 string
362 dot_grid_seq(const rect_element::data& drin, const color klr,
363  const int radius = 80,
364  const int maxwidth = 8, const int maxheight = 4,
365  const int xstart = 40, const int ystart = 100)
366 {
367  // 1920x1080 landscape baselines.
368  // 8 wide, 4 high
369  // 160 pixel diameter, start at x 40, y 100, move 240.
370 
371  static std::mt19937_64 rg(std::random_device{}());
372  auto distw = std::uniform_int_distribution<>(0, maxwidth);
373  auto disth = std::uniform_int_distribution<>(0, maxheight);
374  auto distb = std::uniform_int_distribution<>(0, 1);
375  const int gwidth = distw(rg);
376  const int gheight = disth(rg);
377 
378  // Start at bottom off-screen and go up.
379  // These should all be arguments if this were to be parameterized.
380  size_type negspace = 2 * radius * 1.5;
381  size_type yinit = drin._M_height - radius - ystart;
382 
383  // style sty = { klr, 1.0, klr, 0.0, 0 };
384  style sty = { klr, .9, klr, 0.0, 0 };
385 
386  string ret;
387  for (auto iheight = 0; iheight < gheight; iheight++)
388  {
389  for (auto iwidth = 0; iwidth < gwidth; iwidth++)
390  {
391  const bool b = distb(rg);
392  if (b)
393  {
394  circle_element csvg;
395  size_type x = radius + xstart + (iwidth * negspace);
396  size_type y = yinit - (iheight * negspace);
397  circle_element::data dc = { x, y, radius };
398  csvg.start_element();
399  csvg.add_data(dc);
400  csvg.add_style(sty);
401  csvg.finish_element();
402  ret += csvg.str();
403  }
404  }
405  }
406 
407  return ret;
408 }

411 /**
412  Simulated optical sound dots, or film processing punches,
413  an experimental film stylistic.
414 
415  "The conclusion of a Warhol film is usually announced by the
416  appearance of a flurry of white dots—an artifact of the
417  identification number punched into the last few feet of the roll
418  at the lab." Kyle Westphal, Andy Warhol's Magic Trick
419 
420  Analyzing Warhol's Outer and Inner Space frame-by-frame, assume:
421  1920 x 1080 frame size
422  160 pixel diameter circles, with 240 pixel horiz/vert spacing
423  100 pixel from edge
424 
425  At 30fps, sequence is:
426  - 2 frames dot pattern 1
427  - 2 frames dot pattern 2
428  - 17 frames nothing
429  - 2 frames dot pattern 3
430  - 2 frames dot pattern 4
431 
432  Simplify to:
433  - 2 frames dot pattern 1
434  - 2 frames dot pattern 2
435  - 20 frames blank
436 
437  compositions are no more than four across, and sometimes go off the frame
438 
439  r == frame size
440 */
441 strings
442 optical_sound_dots_seq(const rect_element::data& drin, const color klr,
443  size_t fps = 30, double sec = 1.0,
444  const int radius = 80,
445  const int maxw = 8, const int maxh = 4,
446  const int xstart = 40, const int ystart = 100)
447 {
448  strings ret;
449  size_t framesn = fps * sec; // Calc number of frames needed.
450  size_t step_size = 4 + 17 + 4;
451  for (size_t i = 1; i + step_size < framesn; i += step_size)
452  {
453  string dotpattern1 = dot_grid_seq(drin, klr, radius, maxw, maxh,
454  xstart, ystart);
455  ret.push_back(dotpattern1);
456  string dotpattern2 = dot_grid_seq(drin, klr, radius, maxw, maxh,
457  xstart, ystart);
458  ret.push_back(dotpattern2);
459 
460  string empty;
461  ret.insert(ret.end(), 17, empty);
462 
463  string dotpattern3 = dot_grid_seq(drin, klr, radius, maxw, maxh,
464  xstart, ystart);
465  ret.push_back(dotpattern3);
466  string dotpattern4 = dot_grid_seq(drin, klr, radius, maxw, maxh,
467  xstart, ystart);
468  ret.push_back(dotpattern4);
469  }
470 
471  return ret;
472 }

479 swipe_left_seq(const rect_element::data& drin, const string imgf,
480  const color klr = color::white, size_t fps = 30, double sec = 9)
481 {
482  // Watch for horizontal tearing, tricky.

485  // double sec = 9; // swipesec implies ~640 pix/sec landscape ie 21 pix/frame

488  const auto width = drin._M_width;

491  strings ret;

493  {

496  r.add_data(drin);
497  r.add_style(styl);
498  r.finish_element();

504  img.add_data(di);
505  img.finish_element();
