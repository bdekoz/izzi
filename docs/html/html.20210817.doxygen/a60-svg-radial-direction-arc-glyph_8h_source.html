26 /// Arc + arrow glyph that traces path of start to finish trajectory.
27 void
28 direction_arc_at(svg_element& obj, const point_2t origin,
29  const double rr, svg::style s, const uint spacer = 10)
30 {
31  const double r = rr - spacer;
32  const auto [ mindeg, maxdeg ] = get_radial_range();
33  auto anglemin = zero_angle_north_cw(mindeg);
34  auto anglemax = zero_angle_north_cw(maxdeg);
35 
36  // Define arc.
37  const bool largearcp = abs(anglemax - anglemin) > 180;
38  const bool cwp = true;
39  point_2t p0 = get_circumference_point_d(anglemin, r, origin);
40  point_2t p4 = get_circumference_point_d(anglemax, r, origin);
41  string sarc = make_path_arc_circumference(p0, p4, r, largearcp, cwp);
42 
43  // Adjust style so the stroke color matches the fill color.
44  s._M_stroke_color = s._M_fill_color;
45 
46  auto rspacer = spacer - 2;
47 
48  point_2t p5 = get_circumference_point_d(anglemax, r + rspacer, origin);
49  point_2t p7 = get_circumference_point_d(anglemax, r - rspacer, origin);
50 
51  // Circumference arc length desired is radius times the angle of the
52  // arc.
53  auto theta = 2 * rspacer / r;
54  auto thetad = theta * 180 / k::pi;
55  auto alignd = zero_angle_north_cw(maxdeg + thetad);
56  point_2t p6 = get_circumference_point_d(alignd, r, origin);
57 
58  // Define marker.
59  std::ostringstream ossm;
60  ossm << "M" << k::space << to_string(p4) << k::space;
61  ossm << "L" << k::space;
62  ossm << to_string(p5) << k::space;
63  ossm << to_string(p6) << k::space;
64  ossm << to_string(p7) << k::space;
65  ossm << to_string(p4) << k::space;
66 
67  // Adjust style so that fill will be shown, and stroke hidden.
68  s._M_fill_opacity = 1;
69  s._M_stroke_opacity = 0;
70 
71  // End marker path.
72  path_element pmarker;
73  path_element::data dm = { ossm.str(), 0 };
74  pmarker.start_element();
75  pmarker.add_data(dm);
76  pmarker.add_style(s);
77  pmarker.finish_element();
78  obj.add_element(pmarker);
79 
80  // Reset style.
81  s._M_fill_opacity = 0;
82  s._M_stroke_opacity = 1;
83 
84  // Arc path.
85  path_element parc;
86  path_element::data da = { sarc, 0 };
87  parc.start_element();
88  parc.add_data(da);
89  parc.add_style(s);
90  parc.finish_element();
91  obj.add_element(parc);
92 }

95 /// Title on same arc.
96 void
97 direction_arc_title_at(svg_element& obj, const point_2t origin,
98  const int radius, svg::style s,
99  const string title)
100 {
101  // Make arc text path
102  const auto [ mindeg, maxdeg ] = get_radial_range();
103  auto mina = zero_angle_north_cw(mindeg);
104  auto maxa = zero_angle_north_cw(maxdeg);
105 
106  point_2t pmin = get_circumference_point_d(mina, radius, origin);
107  point_2t pmax = get_circumference_point_d(maxa, radius, origin);
108 
109  string titlearc = make_path_arc_circumference(pmax, pmin, radius);
110 
111  string arc_name("arc-text");
112  path_element parc(false);
113  path_element::data da = { titlearc, 0 };
114  parc.start_element(arc_name);
115  parc.add_data(da);
116  parc.add_style(s);
117  parc.finish_element();
118  obj.add_element(parc);
119 
120  // Make label text and style it.
121  typography typo = k::apercu_typo;
122  typo._M_size = 10;
123  typo._M_anchor = typography::anchor::start;
124  typo._M_align = typography::align::left;
125 
126  // Put it together.
127  text_element::data dt = { 0, 0, title, typo };
128  text_path_element tp(arc_name, "30%");
129  tp.start_element();
130  tp.add_data(dt);
131  tp.finish_element();
132  obj.add_element(tp);
133 }
