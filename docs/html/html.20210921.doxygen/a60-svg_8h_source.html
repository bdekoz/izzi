74 /// Find cartesian distance between two 2D points.
75 space_type
76 distance_cartesian(const point_2t& p1, const point_2t& p2)
77 {
78  auto [ x1, y1 ] = p1;
79  auto [ x2, y2 ] = p2;
80  auto distancex = (x2 - x1) * (x2 - x1);
81  auto distancey = (y2 - y1) * (y2 - y1);
82  space_type distance = sqrt(distancex + distancey);
83  return distance;
84 }

109 /// Conversion between point size to pixels given dpi density.
110 double
111 pt_to_px(const uint i = 1)
112 {
113  // 1pt is equal to exactly 1/72th of an inch.
114  // On a 72dpi output device (display), this is 1 (aka 72 * 1/72).
115  // On a 90dpi output device (display), this is 1.25 (aka 90 * 1/72).
116  // On a 96dpi output device (display), this is 1.33 (aka 96 * 1/72).
117  // constexpr double dpimult = (get_dpi() / 72);
118 
119  // Or 18 pt -> 13.5 px -> 1.33
120  // Or 30 pt -> 22.5 px -> 1.33
121  constexpr double dpimult = 1.33;
122  return std::lround(i * dpimult);
123 }
