43 /// Scale value from min to max on range (nfloor, nceil).
44 double
45 normalize_value_on_range(const size_type value, const size_type min,
46  const size_type max,
47  const size_type nfloor, const size_type nceil)
48 {
49  double rmultp(nceil - nfloor);

194 /// Text of maxlen length, overflow goes on line below.
195 uint
196 text_line_n(svg_element& obj, const point_2t origin, const string text,
197  svg::typography typo, const int sz, const uint maxlen)
198 {
199  auto [ x, y ] = origin;
200 
201  string textcut(text);
202  while (textcut.size() > maxlen)
203  {
204  // Find last space character in the specified maxium range, aka mark.
205  auto sppos = textcut.find_last_of(k::space, maxlen);
206  if (sppos == string::npos)
207  sppos = maxlen;
208  else
209  {
210  // Cut after space (mark).
211  sppos += 1;
212  }
213 
214  string namesubs = textcut.substr(0, sppos);
215  sized_text(obj, typo, sz, namesubs, x, y);
216  textcut = textcut.substr(sppos);
217  y += sz;
218  }
219  sized_text(obj, typo, sz, textcut, x, y);
220  return y;
221 }

224 /// Text of maxlen length rotated, overflow goes on line below.
225 uint
226 text_line_n_r(svg_element& obj, const point_2t origin, const string text,
227  svg::typography typo, const uint sz, const uint maxlen,
228  const uint lettingsz = 0)
229 {
230  const auto [ x, y ] = origin;
231  const double line_max = std::ceil(double(text.size()) / maxlen);
232  const uint lines(line_max);
233 
234  uint xmin(x);
235  string textcut(text);
236 
237  uint linen(0);
238  while (linen < lines)
239  {
240  // Find x offset, with first line being the max length above the
241  // origin, working down. The last line is at the origin.
242  auto xp = x - ((sz + lettingsz) * (lines - linen - 1));
243 
244  size_t epos(0);
245  if (textcut.size() < maxlen)
246  epos = textcut.size();
247  else
248  {
249  // Find last space character in the specified maxium range, aka mark.
250  auto sppos = textcut.find_last_of(k::space, maxlen);
251  if (sppos == string::npos)
252  sppos = maxlen;
253  else
254  {
255  // Cut after space (mark).
256  sppos += 1;
257  }
258  epos = sppos;
259  }
260 
261  string namesubs = textcut.substr(0, epos);
262  sized_text_r(obj, typo, sz, namesubs, xp, y, -90);
263  textcut = textcut.substr(epos);
264  ++linen;
265  xmin = std::min(xmin, uint(xp));
266  }
267  return xmin;
268 }

371 /// Draws a ring centered at origin of radius r, with outer and inner
372 /// radial gradient of blurspace in each direction.
373 /// klr == fade from
374 /// fadeklr == fade to. Background is transparent if none.
375 void
376 point_to_ring_halo(svg_element& obj, const point_2t origin,
377  const space_type radius, const double blurspace,
378  const svg::color klr,
379  const svg::color fadeklr = color::none,
380  const double opacity = 1)
381 {
382  using atype = decltype(obj._M_area)::atype;
383 
384  auto [ xd, yd ] = origin;
385  const atype x(xd);
386  const atype y(yd);
387 
388  // outer ring == upper bound, radius + variance.
389  const double oring = radius + blurspace;
390 
391  // inner ring == lower bound, radius - variance.
392  const double iring = radius - blurspace;
393 
394  // mangled args for name.
395  std::ostringstream oss;
396  oss << "x" << std::to_string(x) << k::hyphen
397  << "y" << std::to_string(y) << k::hyphen
398  << "r" << std::to_string(radius) << k::hyphen
399  << "blurspace" << std::to_string(blurspace);
400  const string mangle(oss.str());
401 
402  // outer
403  // strategy: make a bigger circle cprime, then do a radial gradient to it
404  // starting gradient from color at radius to 100% white/tranparent at cprime.
405  const string rgrado_name(string("radialout") + k::hyphen + mangle);
406  radial_gradient rgrado;
407  rgrado.start_element(rgrado_name);
408  rgrado.stop(rgrado.offset_percentage(radius, oring), fadeklr, 0);
409  rgrado.stop(rgrado.offset_percentage(radius, oring), klr, opacity);
410  rgrado.stop("100%", color::white, 0);
411  rgrado.finish_element();
412  obj.add_element(rgrado);
413 
414  circle_element co;
415  circle_element::data dco = { x, y, atype(oring) };
416  co.start_element();
417  co.add_data(dco);
418  co.add_fill(rgrado_name);
419  co.finish_element();
420  obj.add_element(co);
421 
422  // inner
423  // strategy: make a smaller circle cprime, then do a radial gradient from it
424  // starting gradient from white/transparent at cprime to color at r.
425  const string rgradi_name(string("radialin") + k::hyphen + mangle);
426  radial_gradient rgradi;
427  rgradi.start_element(rgradi_name);
428  rgradi.stop(rgradi.offset_percentage(iring, radius), fadeklr, 0);
429  rgradi.stop("100%", klr, opacity);
430  rgradi.finish_element();
431  obj.add_element(rgradi);
432 
433  // Float/Int conversion and rounding, add one to radius to close gap.
434  circle_element ci;
435  circle_element::data dci = { x, y, radius + 1 };
436  ci.start_element();
437  ci.add_data(dci);
438  ci.add_fill(rgradi_name);
439  ci.finish_element();
440  obj.add_element(ci);
441 }

444 /// Lines radiating from center point (x,y).
445 void
446 point_2d_to_ray(svg_element& obj, double x, double y, svg::style s,
447  space_type r = 4, const uint nrays = 10)
448 {
449  using atype = decltype(obj._M_area)::atype;
450 
451  // End points on the ray.
452  // Pick a random ray, use an angle in the range [0, 2pi].
453  static std::mt19937_64 rg(std::random_device{}());
454  auto distr = std::uniform_real_distribution<>(0.0, 2 * 22/7);
455  auto disti = std::uniform_int_distribution<>(-3, 3);
456 
457  for (uint i = 0; i < nrays; ++i)
458  {
459  double theta = distr(rg);
460  double rvary = disti(rg);
461 
462  atype xe = x + (r + rvary) * std::cos(theta);
463  atype ye = y + (r + rvary) * std::sin(theta);
464 
465  line_element::data dr = { atype(x), xe, atype(y), ye };
466  line_element ray;
467  ray.start_element();
468  ray.add_data(dr);
469  ray.add_style(s);
470  ray.finish_element();
471  obj.add_element(ray);
472  }
473 }

569 /// Points like: get_circumference_point_d(zero_angle_north_cw(0), r, origin)
570 string
571 make_path_arc_closed(const point_2t& origin, const point_2t& start,
572  const point_2t& end, const space_type r,
573  const int arcflag = 0, const int sweepflag = 0)
574 {
575  // Define path as starting at origin, line to circumference point start,
576  // arc to circumfrence point end, line back to origin.
577  // A rx ry x-axis-rotation large-arc-flag sweep-flag x y

587  oss << 0 << k::space << arcflag << k::space << sweepflag << k::space;
588  oss << to_string(end) << k::space;
589  oss << "L" << k::space << to_string(origin) << k::space;

727 /// Embed svg in group element.
728 /// @origin is where glyph placement is inside containing svg element.
729 /// @origsize is original file width/height constant
730 /// @isize is final width/height
731 /// @isvg is the string from one of the two functions above (*_to_svg_insert).
732 /// @syl is overide style information: defaults to no_style.
733 ///
734 /// NB This only works is the file has no styles set in svg, group, or
735 /// individual element definitions (like circle, path, rectangle,
736 /// etc.).
737 ///
738 /// See: https://developer.mozilla.org/en-US/docs/Web/SVG/Element/svg
739 svg_element
740 insert_svg_at(svg_element& obj, const string isvg,
741  const point_2t origin, const double origsize, const double isize,
742  const double angled = 0, const style& styl = k::no_style)
743 {
744  // offset
745  auto [ objx, objy ] = origin;
746  const int x = objx - (isize / 2);
747  const int y = objy - (isize / 2);
748  string xformtranslate(transform::translate(x, y));
749 
750  // scaled
751  string xformscale;
752  if (isize != origsize)
753  {
754  const double scalex(isize / origsize);
755  xformscale = transform::scale(scalex);
756  }
757 
758  // rotated
759  string xformrotate;
760  if (angled != 0)
761  {
762  xformrotate = transform::rotate(90 - angled, objx, objy);
763  }
764 
765  // Order of transformations matters...
766  string ts(xformrotate + k::space + xformtranslate + k::space + xformscale);
767 
768  group_element gsvg;
769  gsvg.start_element("inset svg", transform(), ts, styl);
770  gsvg.add_raw(isvg);
771  gsvg.finish_element();
772  obj.add_element(gsvg);
773 
774  return obj;
775 }
