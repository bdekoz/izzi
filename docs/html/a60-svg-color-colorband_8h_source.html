28/// Specific regions of spectrum as bands of color.
29/// Each band has a starting hue and a number of known good samples.
30/// This is then used to seed a generator that computes more of similar hues.
31using colorband = std::tuple<color, ushort>;
32
33/// Izzi palette-specific offsets for colorbands.
34constexpr colorband cband_bw = std::make_tuple(color::white, 2);
35constexpr colorband cband_gray = std::make_tuple(color::white, 16);
36constexpr colorband cband_y = std::make_tuple(color::hellayellow, 10);
37constexpr colorband cband_r = std::make_tuple(color::red, 17);
38constexpr colorband cband_g = std::make_tuple(color::green, 21);
39constexpr colorband cband_b = std::make_tuple(color::blue, 34);
40constexpr colorband cband_p = std::make_tuple(color::purple, 33);
41constexpr colorband cband_o = std::make_tuple(color::orange, 10);
42constexpr colorband cband_brown = std::make_tuple(color::duboisbrown1, 7);

45/**
46 Generate a color band from starting hue and seeds.
47
48 Algorithm is average two known good, where two picked randomly.
49
50 Return type is a vector of generated color_qi types.
51*/
52color_qis
53make_color_band_v1(const colorband& cb, const ushort neededh)
54{
55 // Find starting hue and number of samples in the color band.
56 color c = std::get<0>(cb);
57 ushort hn = std::get<1>(cb);
58
59 // Find initial offset.
60 auto& spectrum = active_spectrum();
61 auto itr = std::find(spectrum.begin(), spectrum.end(), c);
62 if (itr == spectrum.end())
63 {
64 string m("collection::make_color_band_v1: at the end of " + to_string(c));
65 throw std::runtime_error(m);
66 }
67
68 // Randomness.
69 static std::mt19937_64 rg(std::random_device{}());
70
71 // Setup random picker of sample hues in band.
72 auto disti = std::uniform_int_distribution<>(0, hn - 1);
73
74 // Set up random percentage for combining two colors.
75 auto distr = std::uniform_real_distribution<>(0, 1);
76
77 // Generate new from averaging random samples, cache in return vector.
78 color_qis cband;
79 for (ushort i = 0; i < neededh; ++i)
80 {
81 // New color.
82 ushort o1 = disti(rg);
83 ushort o2 = disti(rg);
84 color_qi c1 = *(itr + o1);
85 color_qi c2 = *(itr + o2);
86
87 // Combine.
88 double c1r = distr(rg);
89 double c2r = 2.0 - c1r;
90 color_qi cgen = combine_color_qi(c1, c1r, c2, c2r);
91 cband.push_back(cgen);
92 }
93
94 return cband;
95}

98/// Algorightm is HSV generation.
99color_qis
100make_color_band_v2(const colorband& cb, const ushort neededh)
101{
102 // Find starting hue and number of samples in the color band.
103 color_qi h = std::get<0>(cb);
104 const ushort hn = std::get<1>(cb) - 1;
105
106 color_qis cband;
107 color_qis cbando;
108 for (ushort i = 0; i < neededh; ++i)
109 {
110 // Start by pushing all original hues into colorband if space.
111 if (i < hn)
112 {
113 cband.push_back(h);
114 h = next_color(h);
115 }
116 else
117 {
118 // All discrete hues in the original colorband; aka palette.
119 if (i == hn)
120 cbando = cband;
121
122 for (ushort ii = 0; ii < cbando.size(); ++ii)
123 {
124 color_qi krgb = cbando[ii];
125 color_qf khsv(krgb);
126 }
127
128 // Done.
129 }
130 }
131
132 return cband;
133}

142/// Flip through color band colors.
143/// @bandn is the number of colors in the colorband.
144color_qi
145next_in_color_band(const colorband& cb, const ushort bandn = 400)
146{
147 // Generate bands.
148 static color_qis gband_bw = make_color_band(cband_bw, bandn);
149 static color_qis gband_y = make_color_band(cband_y, bandn);
150 static color_qis gband_r = make_color_band(cband_r, bandn);
151 static color_qis gband_g = make_color_band(cband_g, bandn);
152 static color_qis gband_b = make_color_band(cband_b, bandn);
153 static color_qis gband_p = make_color_band(cband_p, bandn);
154 static color_qis gband_o = make_color_band(cband_o, bandn);
155 static color_qis gband_brown = make_color_band(cband_brown, bandn);
156
157 color_qi ret;
158 const color c = std::get<0>(cb);
159 switch (c)
160 {
161 case color::white:
162 ret = gband_bw.back();
163 gband_bw.pop_back();
164 break;
165 case color::hellayellow:
166 ret = gband_y.back();
167 gband_y.pop_back();
168 break;
169 case color::orange:
170 ret = gband_o.back();
171 gband_o.pop_back();
172 break;
173 case color::duboisbrown1:
174 ret = gband_brown.back();
175 gband_brown.pop_back();
176 break;
177 case color::red:
178 ret = gband_r.back();
179 gband_r.pop_back();
180 break;
181 case color::green:
182 ret = gband_g.back();
183 gband_g.pop_back();
184 break;
185 case color::blue:
186 ret = gband_b.back();
187 gband_b.pop_back();
188 break;
189 case color::purple:
190 ret = gband_p.back();
191 gband_p.pop_back();
192 break;
193 default:
194 string m("next_in_color_band:: error");
195 m += k::newline;
196 m += "color is: ";
197 m += to_string(c);
198 throw std::runtime_error(m);
199 break;
200 }
201 return ret;
202}
