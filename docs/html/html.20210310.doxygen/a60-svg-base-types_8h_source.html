54 double
55 pt_to_px(uint i)
56 {
57  // 1pt is equal to exactly 1/72th of an inch.
58  // On a 72dpi output device (display), this is 1 (aka 72 * 1/72).
59  // On a 90dpi output device (display), this is 1.25 (aka 90 * 1/72).
60  // On a 96dpi output device (display), this is 1.33 (aka 96 * 1/72).
61  const uint dpi = k::get_dpi();
62  return (i * (dpi / 72));
63 }

269  const std::string
270  add_attribute() const
271  {
272  const std::string name("__name");
273  const std::string size("__size");
274  const std::string align("__align");
275 
276  std::string strip1 = \
277  R"(font-family="__name" font-size="__sizepx" text-align="__align" )";
278 
279  string_replace(strip1, name, _M_face);
280  string_replace(strip1, size, std::to_string(_M_size));
281  string_replace(strip1, align, to_string(_M_align));
282 
283  const std::string weight("__weight");

285 
286  std::string strip2 = \

288 
289  string_replace(strip2, weight, to_string(_M_w));
290  string_replace(strip2, property, to_string(_M_p));
