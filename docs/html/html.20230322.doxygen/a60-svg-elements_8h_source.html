27/// Abstract base class for all SVG Elements.
28struct element_base
29{
30 using stream_type = std::ostringstream;
31
32 // Underlying units for 2D (x,y) mapping (same as area::atype).
33 using atype = space_type; // ... floating point cartesian points
34 //using atype = size_type; // ... integer cartesian points
35
36 /// Virtual, only one buffer.
37 stream_type _M_sstream;
38
39 virtual void
40 start_element() = 0;
41
42 virtual void
43 finish_element() = 0;
44
45 /// Empty when the output buffer is.
46 bool
47 empty() { return _M_sstream.str().empty(); }
48
49 string
50 str() { return _M_sstream.str(); }
51
52 void
53 str(const string& s) { return _M_sstream.str(s); }
54
55 // Add raw string to group, filter, blend/gradient elements.
56 void
57 add_raw(const string& raw)
58 { _M_sstream << k::space << raw << k::newline; }
59
60 void
61 add_fill(const string id)
62 {
63 _M_sstream << k::space;
64 _M_sstream << "fill=" << k::quote;
65 _M_sstream << "url(#" << id << ")" << k::quote;
66 }
67
68 void
69 add_filter(const string id)
70 {
71 _M_sstream << k::space;
72 _M_sstream << "filter=" << k::quote;
73 _M_sstream << "url(#" << id << ")" << k::quote;
74 }
75
76 void
77 add_style(const style& sty)
78 { _M_sstream << to_string(sty); }
79
80 /// Common transforms include rotate(180)
81 void
82 add_transform(const string s)
83 {
84 if (!s.empty())
85 _M_sstream << k::space << "transform=" << k::quote << s << k::quote;
86 }
87};

105 /// For groups of elements that have the same name.
106 ///
107 /// Also, although one can nest SVG elements inside another SVG by
108 /// using an 'image_element', the display quality is lacking in
109 /// inkscape. To work around this, insert the contents of the nested
110 /// SVG into a named group element instead.
111 void
112 start_element(string name)
113 {
114 _M_sstream << "<g id=" << k::quote << name << k::quote
115 << ">" << k::newline;
116 }
117
118 // For groups of elements with style as specified.
119 void
120 start_element(string name, const style& sty)
121 {
122 _M_sstream << "<g id=" << k::quote << name << k::quote << k::space;
123 add_style(sty);
124 _M_sstream << '>' << k::newline;
125 }
126
127 // For groups of elements with transforms and style if specified.
128 void
129 start_element(string name, const transform, const string ts,
130 const style& sty = k::no_style)
131 {
132 _M_sstream << "<g id=" << k::quote << name << k::quote;
133 add_transform(ts);
134
135 // Only add style if it is not the default argument.
136 const string nostr = to_string(color::none);
137 const string fillstr = to_string(sty._M_fill_color);
138 const bool colornonep = nostr == fillstr;
139 if (!colornonep)
140 add_style(sty);
141 _M_sstream << '>' << k::newline;
142 }
143
144 void
145 finish_element();
146};

210 // Some filter effects get clipped when appied to an element's area,
211 // so this allows filters to have an element + filter area instead.
212 void
213 start_element(const string id, const area<> blur_area, const point_2t p)
214 {
215 auto [ width, height ] = blur_area;
216 auto [ x, y ] = p;
217 _M_sstream << "<filter id=" << k::quote << id << k::quote << k::space
218 << "x=" << k::quote << x << k::quote << k::space
219 << "y=" << k::quote << y << k::quote << k::space
220 << "width=" << k::quote << width << k::quote << k::space
221 << "height=" << k::quote << height << k::quote
222 << ">"
223 << k::newline;
224 }
225
226 void
227 finish_element();
228
229 void
230 add_data(const string fltr)
231 { _M_sstream << fltr; }
232
233 // https://drafts.fxtf.org/filter-effects/#elementdef-fegaussianblur
234 // in == SourceGraphic, SourceAlpha, FillPaint, StrokePaint
235 // dev == 1 or 1,1 (0 default if two then x, y direction)
236 string
237 gaussian_blur(string in, string dev)
238 {
239 // <feGaussianBlur in="SourceGraphic" stdDeviation="20" />
240 stream_type stream;
241 stream << "<feGaussianBlur in=";
242 stream << k::quote << in << k::quote << k::space;
243 stream << "stdDeviation=" << k::quote << dev << k::quote << k::space;
244 stream << "/>";
245 return stream.str();
246 }
247
248 string
249 gaussian_blur(string dev)
250 { return gaussian_blur("SourceGraphic", dev); }

288 // off == 10%
289 void
290 stop(const string off, const color& klr, const double opacity = 1.0)
291 {
292 _M_sstream << "<stop offset=" << k::quote << off << k::quote << k::space

306 {

309 return std::to_string(perc) + "%";

381/**
382 Text SVG element.
383
384 Specification reference:
385 https://developer.mozilla.org/en-US/docs/Web/SVG/Element/text
386
387 Attributes:
388 x, y, dx, dy, text-anchor, rotate, textLength, lengthAdjust
389*/
390struct text_element : virtual public element_base
391{
392 struct data
393 {
394 atype _M_x_origin;
395 atype _M_y_origin;
396 string _M_text;
397 typography _M_typo;
398 };
399
400 // So text_path_element can substitute the text_path part without
401 // duplicating the text formatting and style parts....
402 virtual void
403 add_text(string txt)
404 { _M_sstream << txt; }
405
406 /// Either serialize immediately (as below), or create data structure
407 /// that adds data to data_vec and then finish_element serializes.
408 void
409 add_data(const data& d,
410 const string trans = "", const unit utype = svg::unit::point)
411 {
412 const string x("__x");
413 const string y("__y");
414 const string attr("__attr");
415 const string style("__style");
416
417 string strip = R"_delimiter_(x="__x" y="__y" __attr __style)_delimiter_";
418
419 // Add attributes.
420 string_replace(strip, x, std::to_string(d._M_x_origin));
421 string_replace(strip, y, std::to_string(d._M_y_origin));
422 string_replace(strip, attr, d._M_typo.add_attribute(utype));
423 string_replace(strip, style, to_string(d._M_typo._M_style));
424 _M_sstream << strip;
425 add_transform(trans);
426 _M_sstream << '>';
427
428 // Add text data.
429 add_text(d._M_text);
430 }
431
432 void
433 start_element()
434 { _M_sstream << "<text "; }
435
436 void
437 finish_element();
438
439 /// For text list output, use tspan for line breaks. This span
440 /// creates a new horizontal line for every tspan block, starting at
441 /// xpos with spacing dy (1.2em).
442 static string
443 start_tspan_y(uint xpos, string dy)
444 {
445 const string x("__x");
446 const string dys("__dy");
447 string strip = R"_delimiter_(<tspan x="__x" dy="__dy">)_delimiter_";
448 string_replace(strip, x, std::to_string(xpos));
449 string_replace(strip, dys, dy);
450 return strip;
451 }
452
453 static string
454 start_tspan_y(uint xpos, uint dy)
455 { return start_tspan_y(xpos, std::to_string(dy)); }
456
457 /// For text list output, use tspan for line breaks. This span
458 /// creates a new vertical line space for every tspan block, starting
459 /// at xpos with horizontal spacing dx ("1.4em").
460 static string
461 start_tspan_x(uint xpos, string dx)
462 {
463 const string x("__x");
464 const string dxs("__dx");
465 string strip = R"_delimiter_(<tspan x="__x" dx="__dx">)_delimiter_";
466 string_replace(strip, x, std::to_string(xpos));
467 string_replace(strip, dxs, dx);
468 return strip;
469 }
470
471 static string
472 start_tspan_x(uint xpos, uint dx)
473 { return start_tspan_x(xpos, std::to_string(dx)); }
474
475 static string
476 finish_tspan()
477 { return "</tspan>"; }
478};

480void
481text_element::finish_element()
482{ _M_sstream << "</text>" << k::newline; }

484
485/// Make text span.
486string
487make_tspan_y_from_string_by_token(string s, uint xpos, const char token = ' ')
488{
489 string start(text_element::start_tspan_y(xpos, "0.5em"));
490 string ret = start;
491 for (uint i = 0; i < s.size(); ++i)
492 {
493 const char c = s[i];
494 if (c != token)
495 ret += c;
496 else
497 {
498 ret += text_element::finish_tspan();
499 if (i < s.size() - 1)
500 ret += start;
501 }
502 }
503 ret += text_element::finish_tspan();
504 return ret;
505}

666/**
667 Line SVG element.
668
669 Specification reference:
670 https://developer.mozilla.org/en-US/docs/Web/SVG/Element/line
671
672 Attributes:
673 x, y, width, height, xlink:xref, preserveAspectRatio
674*/
675struct line_element : virtual public element_base
676{

681 atype _M_y_begin;

684
685 // Either serialize immediately (as below), or create data structure
686 // that adds data to data_vec and then finish_element serializes.
687 void
688 add_data(const data& d, const string dasharray = "")
689 {
690 const string x1("__x1");
691 const string x2("__x2");
692 const string y1("__y1");
693 const string y2("__y2");
694 const string dash("__darray");
695
696 const bool dashp = !dasharray.empty();
697 string stripf = \
698 R"_delimiter_(x1="__x1" y1="__y1" x2="__x2" y2="__y2")_delimiter_";
699 string stript = \
700 R"_delimiter_(x1="__x1" y1="__y1" x2="__x2" y2="__y2" stroke-dasharray="__darray")_delimiter_";
701
702 string strip = dashp ? stript : stripf;
703 string_replace(strip, x1, std::to_string(d._M_x_begin));
704 string_replace(strip, x2, std::to_string(d._M_x_end));
705 string_replace(strip, y1, std::to_string(d._M_y_begin));
706 string_replace(strip, y2, std::to_string(d._M_y_end));
707
708 if (dashp)
709 string_replace(strip, dash, dasharray);
710 _M_sstream << strip;
711 }

715 { _M_sstream << "<line "; }
