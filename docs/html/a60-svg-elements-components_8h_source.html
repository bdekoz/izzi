49/// SVG element beginning boilerplate for outermost (containing) svg_element.
50/// Variable: unit, x=0, y=0, width, height
51void
52svg_element::start_element()
53{
54 const string start = R"_delimiter_(<?xml version="1.0" encoding="utf-8"?>
55<!DOCTYPE svg PUBLIC "-//W3C//DTD SVG 1.1//EN" "http://www.w3.org/Graphics/SVG/1.1/DTD/svg11.dtd">
56<svg version="1.1"
57 id="svg2" xml:space="preserve"
58 xmlns:dc="http://purl.org/dc/elements/1.1/"
59 xmlns:cc="http://creativecommons.org/ns#"
60 xmlns:svg="http://www.w3.org/2000/svg"
61 xmlns="http://www.w3.org/2000/svg"
62 xmlns:xlink="http://www.w3.org/1999/xlink"
63)_delimiter_";
64
65 const string unit("__unit");
66 const string width("__width");
67 const string height("__height");
68
69 string strip = R"_delimiter_(x="0__unit" y="0__unit"
70width="__width__unit" height="__height__unit"
71viewBox="0 0 __width __height" enable-background="new 0 0 __width __height">
72)_delimiter_";
73
74 string_replace(strip, unit, to_string(_M_unit));
75 string_replace(strip, width, std::to_string(_M_area._M_width));
76 string_replace(strip, height, std::to_string(_M_area._M_height));
77
78 _M_sstream << start;
79 _M_sstream << strip << std::endl;
80}

83/// SVG element for nested svg_element.
84/// Use name, area, unit, typo for viewport frame
85/// @param p origin x,y position
86/// @param desta height, width of destination, iff > _M_area then enlarge.
87void
88svg_element::start_element(const point_2t p, const area destarea,
89 const style& styl)
90{
91 const string id("__id");
92 string start = R"_delimiter_(<svg id="__id" )_delimiter_";
93 string_replace(start, id, _M_name);
94
95 const string unit("__unit");
96 const string width("__width");
97 const string height("__height");
98 const string dwidth("__dwidth");
99 const string dheight("__dheight");
100 const string px("__x");
101 const string py("__y");
102 string strip = R"_delimiter_(viewBox="0 0 __width __height" x="__x__unit" y="__y__unit" width="__dwidth__unit" height="__dheight__unit" )_delimiter_";
103
104 string_replace(strip, unit, to_string(_M_unit));
105 string_replace(strip, width, std::to_string(_M_area._M_width));
106 string_replace(strip, height, std::to_string(_M_area._M_height));
107
108 const auto [ dw, dh ] = destarea;
109 const auto [ x, y ] = p;
110 string_replace(strip, dwidth, std::to_string(dw));
111 string_replace(strip, dheight, std::to_string(dh));
112 string_replace(strip, px, std::to_string(int(x)));
113 string_replace(strip, py, std::to_string(int(y)));
114
115 // Check to make sure stream starts empty.
116 if (!this->empty())
117 {
118 string m("svg_element::start_element error buffer is not empty: " );
119 m += k::newline;
120 m += this->str();
121 m += k::newline;
122 throw std::runtime_error(m);
123 }
124
125 _M_sstream << start;
126 _M_sstream << strip << std::endl;
127
128 // Only add style if it is not the default argument.
129 const string nostr = to_string(color::none);
130 const string fillstr = to_string(styl._M_fill_color);
131 const bool colornonep = nostr == fillstr;
132 if (!colornonep && (styl._M_fill_opacity != 0 || styl._M_stroke_opacity != 0))
133 {
134 add_style(styl);
135 _M_sstream << k::space;
136 }
137 _M_sstream << '>' << k::newline;
138}
