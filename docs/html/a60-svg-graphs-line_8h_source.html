44/// Decompose/split 2D ranges to 1D spaces, perhaps with scaling.
45void
46split_vrange(const vrange& cpoints, vspace& xpoints, vspace& ypoints,
47 const double xscale = 1, const double yscale = 1)
48{
49 for (const auto& [x, y] : cpoints)
50 {
51 xpoints.push_back(x / xscale);
52 ypoints.push_back(y / yscale);
53 }
54}

57/// Union two ranges.
58vrange
59union_vrange(const vrange& r1, const vrange& r2)
60{
61 vrange vr;
62 vr.insert(vr.end(), r1.begin(), r1.end());
63 vr.insert(vr.end(), r2.begin(), r2.end());
64 return vr;
65}

68/// Combine two vranges, combine values, find min/max and return (xmax, ymax)
69/// NB: Assumes zero is min.
70point_2t
71minmax_vrange(const vrange& r1,
72 const double xscale = 1, const double yscale = 1)
73{
74 vspace xpoints;
75 vspace ypoints;
76 split_vrange(r1, xpoints, ypoints, xscale, yscale);
77 sort(xpoints.begin(), xpoints.end());
78 sort(ypoints.begin(), ypoints.end());
79
80 // For x axis, need to insert padding iff axes are scaled down
81 // and/or have values with truncated significant digits.
82 const bool padp(true);
83 if (padp)
84 {
85 const uint pown = 1;
86 const double sigd = pow(10, pown);
87
88 const double dx = xpoints.back();
89 double ix = std::round(dx * sigd) / sigd;
90 if (ix > dx)
91 {
92 xpoints.push_back(ix);
93 std::cout << "x adj: " << k::tab << dx << k::tab << ix << std::endl;
94 }
95
96 const double dy = ypoints.back();
97 uint iy = std::round(dy * sigd) / sigd;
98 if (iy > dy)
99 {
100 ypoints.push_back(iy);
101 std::cout << "y adj: " << k::tab << dy << k::tab << iy << std::endl;
102 }
103 }
104
105 // Find combined ranges, assume zero start.
106 point_2t rangemaxx = std::make_tuple(xpoints.back(), ypoints.back());
107 return rangemaxx;
108}

111/// Truncate double to double with pown signifigant digits.
112vspace
113narrow_vspace(const vspace& points, uint pown)
114{
115 const double sigd = pow(10, pown);
116 vspace npoints;
117 for (const double& d : points)
118 {
119 double dn(d);
120 if (dn > 0)
121 {
122 uint itrunc(dn * sigd);
123 npoints.push_back(itrunc / sigd);
124 }
125 else
126 npoints.push_back(dn);
127 }
128 return npoints;
129}

178/// Return set of paths corresponding to marker shapes with tooltips.
179string
180make_line_graph_markers_tips(const vrange& points, const vrange& cpoints,
181 const graph_rstate& gstate, const double radius,
182 const bool fillp = true)
183{
184 string ret;
185 for (uint i = 0; i < points.size(); i++)
186 {
187 auto [ vx, vy ] = points[i];
188 auto [ cx, cy ] = cpoints[i];
189
190 // Generate displayed tooltip text....
191 string tipstr(gstate.title);
192 tipstr += k::newline;
193 tipstr += std::to_string(static_cast<uint>(vy));
194 tipstr += '%';
195 tipstr += k::comma;
196 tipstr += k::space;
197 tipstr += std::to_string(static_cast<uint>(vx));
198 tipstr += "ms";
199
200 const string& linecap = gstate.sstyle.linecap;
201 const bool roundp = linecap == "round" || linecap == "circle";
202 const bool squarep = linecap == "square";
203 const bool trianglep = linecap == "triangle";
204
205 // Markers are closed paths that are filled with no stroke.
206 style styl = gstate.lstyle;
207 if (fillp)
208 {
209 styl._M_stroke_opacity = 0;
210 styl._M_fill_opacity = 1;
211 }
212
213 // Circle Centered.
214 // svg::circle_element c = make_circle(cpoints[i], gstate.lstyle, r);
215 if (roundp)
216 {
217 circle_element c;
218 circle_element::data dc = { cx, cy, radius };
219 c.start_element();
220 c.add_data(dc);
221 c.add_style(styl);
222 c.add_raw(element_base::finish_hard);
223 c.add_title(tipstr);
224 c.add_raw(string { circle_element::pair_finish_tag } + k::newline);
225 ret += c.str();
226 }
227
228 // Square Centered.
229 // svg::rect_element r = (cpoints[i], gstate.lstyle, {2 * r, 2 * r});
230 if (squarep)
231 {
232 rect_element r;
233 rect_element::data dr = { cx - radius, cy - radius,
234 2 * radius, 2 * radius };
235 r.start_element();
236 r.add_data(dr);
237 r.add_style(styl);
238 r.add_raw(element_base::finish_hard);
239 r.add_title(tipstr);
240 r.add_raw(string { rect_element::pair_finish_tag } + k::newline);
241 ret += r.str();
242 }
243
244 // Triangle Centered.
245 // svg::path_element t = (cpoints[i], gstate.lstyle, {2 * r, 2 * r});
246 if (trianglep)
247 {
248 path_element p = make_path_triangle(cpoints[i], styl, radius, 120,
249 false);
250 p.add_title(tipstr);
251 p.add_raw(string { path_element::pair_finish_tag } + k::newline);
252 ret += p.str();
253 }
254
255 // Throw if marker style not supported.
256 if (!roundp && !squarep && !trianglep)
257 {
258 string m("make_line_graph_markers_tips:: ");
259 m += "linecap value invalid or missing, currently set to: ";
260 m += linecap;
261 m += k::newline;
262 throw std::runtime_error(m);
263 }
264 }
265 return ret;
266}

269/// Axis Labels
270/// Axis X/Y Ticmarks
271svg_element
272make_line_graph_annotations(const area<> aplate,
273 const vrange& points,
274 const graph_rstate& gstate,
275 const double xscale = 1, const double yscale = 1,
276 const typography typo = k::apercu_typo)
277{
278 using namespace std;
279 svg_element lanno(gstate.title, "line graph annotation", aplate, false);
280
281 // Locate graph area on plate area.
282 auto [ pwidth, pheight ] = aplate;
283 double gwidth = pwidth - (2 * gstate.marginx);
284 double gheight = pheight - (2 * gstate.marginy);
285 const double chartyo = pheight - gstate.marginy;
286 const double chartxe = pwidth - gstate.marginx;
287
288 // Base typo for axis.
289 typography anntypo = typo;
290 anntypo._M_style = k::wcaglg_style;
291 anntypo._M_size = graph_rstate::th1sz;
292
293 // Axes and Labels
294 if (gstate.is_visible(select::axis))
295 {
296 lanno.add_raw(group_element::start_group("axes-" + gstate.title));
297
298 // Add axis labels.
299 point_2t xlabelp = make_tuple(pwidth / 2, chartyo + (gstate.marginy / 2));
300 styled_text(lanno, gstate.xlabel, xlabelp, anntypo);
301
302 point_2t ylabelp = make_tuple(gstate.marginx / 2, pheight / 2);
303 styled_text(lanno, gstate.ylabel, ylabelp, anntypo);
304
305 // Add axis lines.
306 line_element lx = make_line({gstate.marginx, chartyo}, {chartxe, chartyo},
307 gstate.lstyle);
308 line_element ly = make_line({gstate.marginx, chartyo},
309 {gstate.marginx, gstate.marginy},
310 gstate.lstyle);
311 lanno.add_element(lx);
312 lanno.add_element(ly);
313
314 lanno.add_raw(group_element::finish_group());
315 }
316
317 // Base typo for tic labels.
318 // NB: Assume pointsx/pointsy are monotonically increasing.
319 anntypo._M_size = graph_rstate::tticsz;
320 if (gstate.is_visible(select::ticks))
321 {
322 lanno.add_raw(group_element::start_group("ticks-" + gstate.title));
323
324 // Separate tic label values for x/y axis.
325 vspace pointsx;
326 vspace pointsy;
327 split_vrange(points, pointsx, pointsy, xscale, yscale);
328 auto [ maxx, maxy ] = minmax_vrange(points, xscale, yscale);
329 pointsx.push_back(maxx);
330 pointsy.push_back(maxy);
331
332 // X tic labels
333 auto minx = 0;
334 const double xrange(maxx - minx);
335 const double xscale(gwidth / xrange);
336
337 const double ygo = gstate.marginy + gheight + graph_rstate::th1sz;
338
339 // Filter tic labels to unique smallest subset of significant labels.
340 vspace xpointsn = narrow_vspace(pointsx, gstate.xtickdigits);
341
342 // Add x tic labels.
343 std::set<double> xpoints(xpointsn.begin(), xpointsn.end());
344 for (const double& x: xpoints)
345 {
346 //const uint xui = x * xticsteps;
347 const double xto = gstate.marginx + (x * xscale);
348
349 ostringstream oss;
350 oss << fixed << setprecision(gstate.xtickdigits) << x;
351 const string sxui = oss.str() + gstate.xticku;
352
353 styled_text(lanno, sxui, {xto, ygo}, anntypo);
354 }
355
356 // Y tic labels
357 auto miny = 0;
358 const double yrange(maxy - miny);
359 const double yscale(gheight / yrange);
360
361 // Positions for left and right y-axis tic labels.
362 const double xgol = gstate.marginx - graph_rstate::th1sz; // left
363 const double xgor = gstate.marginx + gwidth + graph_rstate::th1sz;// right
364
365 // Filter tic labels to unique smallest subset of significant
366 // lables of yticsteps.
367 const double yticsteps = 10; // number of y tic labels
368 const double ydelta = yrange / yticsteps;
369
370 auto ynarrowl = [ydelta](const double& d)
371 { return static_cast<uint>(d / ydelta); };
372 vector<uint> ypointsi(pointsy.size());
373 std::transform(pointsy.begin(), pointsy.end(), ypointsi.begin(),
374 ynarrowl);
375 std::set<double> ypoints(ypointsi.begin(), ypointsi.end());
376
377 // Add y tic labels.
378 for (const double& y: ypoints)
379 {
380 const uint yui = y * ydelta;
381 const double yto = chartyo - (yui * yscale);
382 const string syui = std::to_string(yui) + gstate.yticku;
383 styled_text(lanno, syui, {xgol, yto}, anntypo);
384 styled_text(lanno, syui, {xgor, yto}, anntypo);
385 }
386
387 lanno.add_raw(group_element::finish_group());
388 }
389
390 return lanno;
391}

394/// Returns a svg_element with the chart and labels
395/// Assume:
396/// vgrange x axis is monotonically increasing
397svg_element
398make_line_graph(const svg::area<> aplate, const vrange& points,
399 const graph_rstate& gstate,
400 const point_2t xrange, const point_2t yrange)
401{
402 using namespace std;
403 svg_element lgraph(gstate.title, "line graph", aplate, false);
404
405 auto [ minx, maxx ] = xrange;
406 auto [ miny, maxy ] = yrange;
407
408 // Locate graph area on plate area.
409 // aplate is total plate area with margins, aka
410 // pwidth = marginx + gwidth + marginx
411 // pheight = marginy + gheight + marginy
412 auto [ pwidth, pheight ] = aplate;
413 double gwidth = pwidth - (2 * gstate.marginx);
414 double gheight = pheight - (2 * gstate.marginy);
415 const double chartyo = pheight - gstate.marginy;
416
417 // Transform data points to scaled cartasian points in graph area.
418 vrange cpoints;
419 for (uint i = 0; i < points.size(); i++)
420 {
421 const point_2t& pt = points[i];
422 auto [ vx, vy ] = pt;
423
424 // At bottom of graph.
425 const double xlen = scale_value_on_range(vx, minx, maxx, 0, gwidth);
426 double x = gstate.marginx + xlen;
427
428 // Y axis grows up from chartyo.
429 const double ylen = scale_value_on_range(vy, miny, maxy, 0, gheight);
430 double y = chartyo - ylen;
431
432 cpoints.push_back(make_tuple(x, y));
433 }
434
435 // Plot path of points on cartesian plane.
436 // Grouped tooltips have to be the last, aka top layer of SVG to work (?).
437 //constexpr ushort line_strategy = line_1_polyline;
438 constexpr ushort line_strategy = line_2_polyline_tooltips;
439 if (gstate.is_visible(select::vector))
440 {
441 if constexpr(line_strategy == line_1_polyline)
442 {
443 // Use polylines and markerspoints
444 polyline_element pl1 = make_polyline(cpoints, gstate.lstyle,
445 gstate.sstyle);
446 lgraph.add_element(pl1);
447 }
448 if constexpr(line_strategy == line_2_polyline_tooltips)
449 {
450 // Use polyline base and set of marker paths with orignal values
451 // as tooltips on top.
452 lgraph.add_raw(group_element::start_group("polyline-" + gstate.title));
453 polyline_element pl1 = make_polyline(cpoints, gstate.lstyle,
454 gstate.sstyle);
455 lgraph.add_element(pl1);
456 lgraph.add_raw(group_element::finish_group());
457
458 lgraph.add_raw(group_element::start_group("values-" + gstate.title));
459 string markers = make_line_graph_markers_tips(points, cpoints,
460 gstate, 3);
461 lgraph.add_raw(markers);
462 lgraph.add_raw(group_element::finish_group());
463 }
464 }
465
466 return lgraph;
467}
