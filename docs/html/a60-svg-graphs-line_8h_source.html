94/// Simplify sorted vrange by removing interior duplicates.
95vrange
96find_change_points(const vrange& vr)
97{
98 // Transform range to the simplest expression, where multiple points
99 // without significant vertical change are coalesed to starting
100 // control point and ending control point.
101 vrange simplevr;
102 point_2t last = { -1.0, -1.0 };
103 double duprangep(false);
104 for (const point_2t& pt : vr)
105 {
106 auto [ x, y] = pt;
107 if (y != get<1>(last))
108 {
109 if (duprangep == true)
110 {
111 simplevr.push_back(last);
112 duprangep = false;
113 }
114 simplevr.push_back(pt);
115 }
116 else
117 duprangep = true;
118 last = pt;
119 }
120 return simplevr;
121}

124/// Simplify change points to only markers where visual complete changes
125/// @param points already simplified change points
126vrange
127find_visual_change_points(const vrange& points)
128{
129 vrange simplest;
130 point_2t last = { -1.0, -1.0 };
131 for (const point_2t& pt : points)
132 {
133 auto [ x, y] = pt;
134 if (y != get<1>(last))
135 simplest.push_back(pt);
136 last = pt;
137 }
138 return simplest;
139}

142/// Return set of images for image tooltips, one for each point.
143/// @param aimg is size of image embedded inside svg element.
144/// @pathprefix is the path to the directory with the store of images
145/// @idimgbase is the root name for what will be document level unique names of
146/// sequentially numbered images (say fximage-, for fximage-101 et al)
147/// Expected, zero filled imageid.
148/// 2025-06-26-android-15-ptablet-talkback-4usted-firefox_13188.webp
149string
150make_line_graph_image_set(const area<> aimg, const vrange& points,
151 const string imgidbase, const string pathprefix,
152 const string imgprefix, const string imgext)
153{
154 string ret;
155 for (const point_2t p : points)
156 {
157 std::ostringstream oss;
158 oss << std::setfill('0') << std::setw(5);
159 oss << static_cast<uint>(std::get<0>(p));
160 const string xms = oss.str();
161
162 const string isrc = pathprefix + imgprefix + xms + imgext;
163 const string imgid = imgidbase + xms;
164 auto [ width, height ] = aimg;
165
166 image_element i;
167 image_element::data di = { isrc, 0, 0, width, height };
168 i.start_element(imgid);
169 i.add_data(di, "hidden", "anonymous");
170 i.finish_element();
171 ret += i.str();
172 }
173 return ret;
174}

176/// Return set of paths of marker shapes with text tooltips.
177string
178make_line_graph_markers(const vrange& points, const vrange& cpoints,
179 const graph_rstate& gstate, const double radius,
180 const string imgidbase = "")
181{
182 string ret;
183 for (uint i = 0; i < points.size(); i++)
184 {
185 auto [ vx, vy ] = points[i];
186 auto [ cx, cy ] = cpoints[i];
187
188 std::ostringstream oss;
189 oss << std::setfill('0') << std::setw(5);
190 oss << static_cast<uint>(vx);
191 const string xms = oss.str();
192 const string imgid = imgidbase + xms;
193
194 // Generate displayed tooltip text....
195 string tipstr(gstate.title);
196 tipstr += k::newline;
197 tipstr += std::to_string(static_cast<uint>(vy));
198 tipstr += '%';
199 tipstr += k::comma;
200 tipstr += k::space;
201 tipstr += std::to_string(static_cast<uint>(vx));
202 tipstr += "ms";
203
204 const string& linecap = gstate.sstyle.linecap;
205 const bool roundp = linecap == "round" || linecap == "circle";
206 const bool squarep = linecap == "square";
207 const bool trianglep = linecap == "triangle";
208
209 // Markers default to closed paths that are filled with no stroke.
210 // Setting visible to vector | echo induces outline behavior.
211 style styl = gstate.lstyle;
212 styl._M_fill_opacity = 1;
213 if (gstate.is_visible(select::echo))
214 styl._M_fill_color = color::white;
215
216
217 // Circle Centered.
218 // svg::circle_element c = make_circle(cpoints[i], gstate.lstyle, r);
219 if (roundp)
220 {
221 circle_element c;
222 circle_element::data dc = { cx, cy, radius };
223 c.start_element();
224 c.add_data(dc);
225 c.add_style(styl);
226 if (!imgidbase.empty())
227 c.add_raw(script_element::tooltip_attribute(imgid));
228 c.add_raw(element_base::finish_tag_hard);
229 c.add_title(tipstr);
230 c.add_raw(string { circle_element::pair_finish_tag } + k::newline);
231 ret += c.str();
232 }
233
234 // Square Centered.
235 // svg::rect_element r = (cpoints[i], gstate.lstyle, {2 * r, 2 * r});
236 if (squarep)
237 {
238 rect_element r;
239 rect_element::data dr = { cx - radius, cy - radius,
240 2 * radius, 2 * radius };
241 r.start_element();
242 r.add_data(dr);
243 r.add_style(styl);
244 if (!imgidbase.empty())
245 r.add_raw(script_element::tooltip_attribute(imgid));
246 r.add_raw(element_base::finish_tag_hard);
247 r.add_title(tipstr);
248 r.add_raw(string { rect_element::pair_finish_tag } + k::newline);
249 ret += r.str();
250 }
251
252 // Triangle Centered.
253 if (trianglep)
254 {
255 string xattr;
256 if (!imgidbase.empty())
257 xattr = script_element::tooltip_attribute(imgid);
258
259 // Visual weight of triangle is smaller, so enlarge slightly.
260 const double tradius = radius * 1.3;
261 path_element p = make_path_triangle(cpoints[i], styl, tradius, 120,
262 false, xattr);
263 p.add_title(tipstr);
264 p.add_raw(string { path_element::pair_finish_tag } + k::newline);
265 ret += p.str();
266 }
267
268 // Throw if marker style not supported.
269 if (!roundp && !squarep && !trianglep)
270 {
271 string m("make_line_graph_markers:: ");
272 m += "linecap value invalid or missing, currently set to: ";
273 m += linecap;
274 m += k::newline;
275 throw std::runtime_error(m);
276 }
277 }
278 return ret;
279}

282/// Axis Labels
283/// Axis X/Y Ticmarks
284/// X line increments
285///
286/// @param aplate = total size of graph area
287/// @param points = vector of {x,y} points to graph
288/// @param gstate = graph render state
289/// @param xscale = scale x axis by this ammount (1000 if converting ms to s)
290/// @param yscale = scale y axis by this ammount
291/// @param typo = typography to use for labels
292svg_element
293make_line_graph_annotations(const area<> aplate,
294 const vrange& points,
295 const graph_rstate& gstate,
296 const double xscalein = 1, const double yscalein = 1,
297 const typography typo = k::apercu_typo)
298{
299 using namespace std;
300 svg_element lanno(gstate.title, "line graph annotation", aplate, false);
301
302 // Locate graph area on plate area.
303 auto [ pwidth, pheight ] = aplate;
304 double gwidth = pwidth - (2 * gstate.xmargin);
305 double gheight = pheight - (2 * gstate.ymargin);
306 const double chartyo = pheight - gstate.ymargin;
307 const double chartxo = gstate.xmargin;
308 const double chartxe = pwidth - gstate.xmargin;
309
310 // Base typo for annotations.
311 typography anntypo = typo;
312 anntypo._M_style = k::wcagg_style;
313 anntypo._M_size = graph_rstate::th1sz;
314
315 // Axes and Labels
316 if (gstate.is_visible(select::axis))
317 {
318 lanno.add_raw(group_element::start_group("axes-" + gstate.title));
319
320 // Add axis labels.
321 point_2t xlabelp = make_tuple(pwidth / 2, chartyo + (gstate.ymargin / 2));
322 styled_text(lanno, gstate.xlabel, xlabelp, anntypo);
323
324 point_2t ylabelp = make_tuple(chartxo / 2, pheight / 2);
325 styled_text(lanno, gstate.ylabel, ylabelp, anntypo);
326
327 // Add axis lines.
328 line_element lx = make_line({chartxo, chartyo}, {chartxe, chartyo},
329 gstate.lstyle);
330 line_element ly = make_line({chartxo, chartyo}, {chartxo, gstate.ymargin},
331 gstate.lstyle);
332 lanno.add_element(lx);
333 lanno.add_element(ly);
334
335 lanno.add_raw(group_element::finish_group());
336 }
337
338 // Base typo for tic labels.
339 // NB: Assume pointsx/pointsy are monotonically increasing.
340 anntypo._M_size = graph_rstate::tticsz;
341 anntypo._M_baseline = typography::baseline::central;
342
343 // Separate tic label values for each (x, y) axis, find ranges for each.
344 auto [ maxx, maxy ] = max_vrange(points, gstate.xticdigits, xscalein, yscalein);
345 auto minx = 0;
346 auto miny = 0;
347
348 const double xrange(maxx - minx);
349 const double xscale(gwidth / xrange);
350 const double yrange(maxy - miny);
351 const double yscale(gheight / yrange);
352
353 // Derive the number of tick marks.
354
355 // Use a multiple of 5 to make natural counting easier.
356 // Start with an assumption of 20 tic marks for the x axis.
357 double xtickn(xrange * 2); // .5 sec
358 if (xtickn < 10)
359 xtickn = 10;
360 if (xtickn > 26)
361 xtickn = xrange;
362
363 // X axis is seconds, xtickn minimum delta is 0.1 sec.
364 double xdelta = std::max(xrange / xtickn, 0.1);
365
366 // Round up to significant digits, so if xdelta is 0.18 round to 0.2.
367 xdelta = std::round(xdelta * gstate.xticdigits * 10) / (gstate.xticdigits * 10);
368
369 // Y axis is simpler, 0, 10, 20, ..., 80, 90, 100 in percent.
370 const double ydelta = yrange / gstate.yticdigits;
371
372 // Generate tic marks
373 const double ygo = gstate.ymargin + gheight + graph_rstate::th1sz;
374 if (gstate.is_visible(select::ticks))
375 {
376 // X tic labels
377 lanno.add_raw(group_element::start_group("tic-x-" + gstate.title));
378 for (double x = minx; x < maxx; x += xdelta)
379 {
380 const double xto = chartxo + (x * xscale);
381 ostringstream oss;
382 oss << fixed << setprecision(gstate.xticdigits) << x;
383 const string sxui = oss.str() + gstate.xticu;
384 styled_text(lanno, sxui, {xto, ygo}, anntypo);
385 }
386 lanno.add_raw(group_element::finish_group());
387
388 // Y tic labels
389 // Positions for left and right y-axis tic labels.
390 lanno.add_raw(group_element::start_group("tic-y-" + gstate.title));
391 const double yticspacer = graph_rstate::th1sz * 2;
392 const double xgol = gstate.xmargin - yticspacer; // left
393 const double xgor = gstate.xmargin + gwidth + yticspacer; // right
394 const double starty = miny != 0 ? miny : miny + ydelta; // skip zero label
395 for (double y = starty; y < maxy + ydelta; y += ydelta)
396 {
397 const double yto = chartyo - (y * yscale);
398 const string syui = std::to_string(static_cast<uint>(y)) + gstate.yticu;
399 styled_text(lanno, syui, {xgol, yto}, anntypo);
400 styled_text(lanno, syui, {xgor, yto}, anntypo);
401 }
402 lanno.add_raw(group_element::finish_group());
403 }
404
405 // Horizontal lines linking left and right y-axis tic label value to each other,
406 // perhaps with magnification-ready micro text.
407 if (gstate.is_visible(select::linex))
408 {
409 lanno.add_raw(group_element::start_group("tic-y-lines-" + gstate.title));
410
411 style hlstyl = gstate.lstyle;
412 hlstyl._M_stroke_color = color::gray10;
413
414 anntypo._M_size = 3;
415 //anntypo._M_style.set_colors(color::gray20);
416 for (double y = miny + ydelta; y < maxy + ydelta; y += ydelta)
417 {
418 // Base line layer.
419 const double yto = chartyo - (y * yscale);
420 line_element lxe = make_line({chartxo + graph_rstate::th1sz, yto},
421 {chartxe - graph_rstate::th1sz, yto}, hlstyl);
422 lanno.add_element(lxe);
423
424 // Add y-axis tic numbers along line for use when magnified.
425 if (gstate.is_visible(select::alt))
426 {
427 // Skip first and last as covered by either Y-axes tic marks.
428 for (double x = minx + xdelta; x < maxx - xdelta; x += xdelta)
429 {
430 const double xto = chartxo + (x * xscale);
431 const string syui = std::to_string(static_cast<uint>(y)) + gstate.yticu;
432 styled_text(lanno, syui, {xto, yto}, anntypo);
433 }
434 }
435 }
436
437 lanno.add_raw(group_element::finish_group());
438 }
439
440 return lanno;
441}

444/// Returns a svg_element with the rendered line graph (char).
445/// Assumptions:
446/// vgrange x axis is monotonically increasing
447///
448/// NB1: Axes and labels drawn in a separate pass (make_line_graph_annotations).
449/// NB2: Output file of x-axis point values for image tooltips if strategy = 3.
450///
451/// @param aplate = total size of graph area
452/// @param points = vector of {x,y} points to graph
453/// @param gstate = graph render state
454/// @param xrange = unified x-axis range for all graphs if multiplot
455/// @param yrange = unified y-axis range for all graphs if multiplot
456/// @param line_strategy = type of graph to render, from simplest to tooltips
457/// @param imgidbase = unique text prefix over multimap for image tooltips
458svg_element
459make_line_graph(const svg::area<> aplate, const vrange& points,
460 const graph_rstate& gstate,
461 const point_2t xrange, const point_2t yrange)
462{
463 using namespace std;
464
465 auto [ minx, maxx ] = xrange;
466 auto [ miny, maxy ] = yrange;
467
468 // Locate graph area on plate area.
469 // aplate is total plate area with margins, aka
470 // pwidth = xmargin + gwidth + xmargin
471 // pheight = ymargin + gheight + ymargin
472 auto [ pwidth, pheight ] = aplate;
473 double gwidth = pwidth - (2 * gstate.xmargin);
474 double gheight = pheight - (2 * gstate.ymargin);
475 const double chartyo = pheight - gstate.ymargin;
476
477 // Transform data points to scaled cartasian points in graph area.
478 vrange cpoints;
479 for (uint i = 0; i < points.size(); i++)
480 {
481 const point_2t& pt = points[i];
482 auto [ vx, vy ] = pt;
483
484 // At bottom of graph.
485 const double xlen = scale_value_on_range(vx, minx, maxx, 0, gwidth);
486 double x = gstate.xmargin + xlen;
487
488 // Y axis grows up from chartyo.
489 const double ylen = scale_value_on_range(vy, miny, maxy, 0, gheight);
490 double y = chartyo - ylen;
491
492 cpoints.push_back(make_tuple(x, y));
493 }
494
495 // Plot path of points on cartesian plane.
496 svg_element lgraph(gstate.title + "_line_graph", "line graph", aplate, false);
497 if (gstate.is_visible(select::vector))
498 {
499 if (gstate.line_mode == chart_line_style_1)
500 {
501 // Use polyline and CSS-based markerspoints all in one line.
502 lgraph.add_raw(group_element::start_group("polyline-" + gstate.title));
503 polyline_element pl1 = make_polyline(cpoints, gstate.lstyle,
504 gstate.sstyle);
505 lgraph.add_element(pl1);
506 lgraph.add_raw(group_element::finish_group());
507 }
508 if (gstate.line_mode == chart_line_style_2)
509 {
510 // Use polyline base on bottom line.
511 // Use a second set of marker paths with value as text tooltips on top.
512 lgraph.add_raw(group_element::start_group("polyline-" + gstate.title));
513 stroke_style no_markerstyle = gstate.sstyle;
514 no_markerstyle.markerspoints = "";
515 polyline_element pl1 = make_polyline(cpoints, gstate.lstyle,
516 no_markerstyle);
517 lgraph.add_element(pl1);
518 lgraph.add_raw(group_element::finish_group());
519
520 // Markers + text tooltips.
521 lgraph.add_raw(group_element::start_group("markers-" + gstate.title));
522 string markers = make_line_graph_markers(points, cpoints, gstate, 3);
523 lgraph.add_raw(markers);
524 lgraph.add_raw(group_element::finish_group());
525 }
526 if (gstate.line_mode == chart_line_style_3)
527 {
528 // Use polyline base on bottom line.
529 // Use a second set of simplified marker paths with value as tooltips on top.
530 // Use text and image tooltips.
531 lgraph.add_raw(group_element::start_group("polyline-" + gstate.title));
532 stroke_style no_markerstyle = gstate.sstyle;
533 no_markerstyle.markerspoints = "";
534 polyline_element pl1 = make_polyline(cpoints, gstate.lstyle,
535 no_markerstyle);
536 lgraph.add_element(pl1);
537 lgraph.add_raw(group_element::finish_group());
538
539 // Markers + text tooltips, add image id + js to make image visible.
540 // Use simplified points, aka only the visual change points.
541 const vrange& vizpoints = find_visual_change_points(points);
542 const vrange& cvizpoints = find_visual_change_points(cpoints); // xxx ?
543
544 lgraph.add_raw(group_element::start_group("markers-" + gstate.title));
545 string markers = make_line_graph_markers(vizpoints, cvizpoints, gstate, 3,
546 gstate.tooltip_id);
547 lgraph.add_raw(markers);
548 lgraph.add_raw(group_element::finish_group());
549
550 // Add images with image id, default to hidden
551 // (make_line_graph_image_set)
552
553 // Add javascript to control images
554 // (script_element::tooltip_script)
555 }
556 }
557
558 return lgraph;
559}
