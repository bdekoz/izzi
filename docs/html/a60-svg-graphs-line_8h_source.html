178/// Create a set of markers bounded by a rectangle of size n.
179string
180make_marker_set_n(const double i)
181{
182 // 4 / 2, etc.
183 const double h(i/2);
184 const string si = std::to_string(static_cast<uint>(i));
185 auto m1 = make_marker_circle("c" + si + "red", {i, i}, {h, h}, h, style_r);
186 auto m2 = make_marker_circle("c" + si + "wcaglg", {i, i}, {h, h}, h, style_wcaglg);
187 auto m3 = make_marker_circle("c" + si + "wcagdg", {i, i}, {h, h}, h, style_wcagdg);
188 auto m4 = make_marker_circle("c" + si + "black", {i, i}, {h, h}, h, b_style);
189 auto m5 = make_marker_triangle("t" + si + "black", {i, i}, {h, h}, h, b_style);
190 auto m6 = make_marker_triangle("t" + si + "wcagg", {i, i}, {h, h}, h, style_wcagg);
191 auto m7 = make_marker_x("x" + si + "wcagg", {i, i}, {h, h}, h, style_wcaglg);
192 auto m8 = make_marker_rect("r" + si + "wcaglg", {i, i}, {h, h}, style_wcaglg);
193 auto m9 = make_marker_rect("r" + si + "wcagdg", {i, i}, {h, h}, style_wcagdg);
194
195 std::ostringstream oss;
196 oss << m1.str() << m2.str() << m3.str() << m4.str()
197 << m5.str() << m6.str() << m7.str() << m8.str() << m9.str();
198 return oss.str();
199}

221/// Return set of paths corresponding to marker shapes with tooltips.
222string
223make_line_graph_markers_tips(const vrange& points, const vrange& cpoints,
224 const graph_rstate& gstate, const double radius)
225{
226 const string finish_hard(string { element_base::finish_tag } + k::newline);
227
228 string ret;
229 for (uint i = 0; i < points.size(); i++)
230 {
231 auto [ vx, vy ] = points[i];
232 auto [ cx, cy ] = cpoints[i];
233
234 // Generate displayed tooltip text....
235 string tipstr(gstate.title);
236 tipstr += k::newline;
237 tipstr += std::to_string(static_cast<uint>(vy));
238 tipstr += '%';
239 tipstr += k::comma;
240 tipstr += k::space;
241 tipstr += std::to_string(static_cast<uint>(vx));
242 tipstr += "ms";
243
244 const bool roundp = gstate.sstyle.linecap == "round";
245 const bool squarep = gstate.sstyle.linecap == "square";
246
247 // Centered.
248 // svg::circle_element c = make_circle(cpoints[i], gstate.lstyle, r);
249 if (roundp)
250 {
251 circle_element c;
252 circle_element::data dc = { cx, cy, radius };
253 c.start_element();
254 c.add_data(dc);
255 c.add_style(gstate.lstyle);
256 c.add_raw(finish_hard);
257 c.add_title(tipstr);
258 c.add_raw(string { circle_element::tag_closing } + k::newline);
259 ret += c.str();
260 }
261
262 // Centered.
263 // svg::rect_element r = (cpoints[i], gstate.lstyle, {2 * r, 2 * r});
264 if (squarep)
265 {
266 rect_element r;
267 rect_element::data dr = { cx - radius, cy - radius,
268 2 * radius, 2 * radius };
269 r.start_element();
270 r.add_data(dr);
271 r.add_style(gstate.lstyle);
272 r.add_raw(finish_hard);
273 r.add_title(tipstr);
274 r.add_raw(string { rect_element::tag_closing } + k::newline);
275 ret += r.str();
276 }
277
278 if (!roundp && !squarep)
279 throw std::runtime_error("make_line_graph_markers_tips linecap missing");
280 }
281 return ret;
282}

285/// Axis Labels
286/// Axis X/Y Ticmarks
287void
288make_line_graph_annotations(const area<> aplate,
289 const vspace& pointsx, const vspace& pointsy,
290 const graph_rstate& gstate,
291 svg_element& lgraph,
292 const typography typo = k::apercu_typo)
293{
294 using namespace std;
295 const string finish_hard(string { element_base::finish_tag } + k::newline);
296
297 // Locate graph area on plate area.
298 auto [ pwidth, pheight ] = aplate;
299 double gwidth = pwidth - (2 * gstate.marginx);
300 double gheight = pheight - (2 * gstate.marginy);
301 const double chartyo = pheight - gstate.marginy;
302 const double chartxe = pwidth - gstate.marginx;
303
304 // Base typo for axis.
305 typography anntypo = typo;
306 anntypo._M_style = style_wcaglg;
307 anntypo._M_size = asz;
308 if (gstate.is_visible(select::axis))
309 {
310 lgraph.add_raw(group_element::start_group("axes-" + gstate.title));
311
312 // Add axis labels.
313 point_2t xlabelp = make_tuple(pwidth / 2, chartyo + (gstate.marginy / 2));
314 styled_text(lgraph, gstate.xlabel, xlabelp, anntypo);
315
316 point_2t ylabelp = make_tuple(gstate.marginx / 2, pheight / 2);
317 styled_text(lgraph, gstate.ylabel, ylabelp, anntypo);
318
319 // Add axis lines.
320 line_element lx = make_line({gstate.marginx, chartyo}, {chartxe, chartyo},
321 gstate.lstyle);
322 line_element ly = make_line({gstate.marginx, chartyo},
323 {gstate.marginx, gstate.marginy},
324 gstate.lstyle);
325 lgraph.add_element(lx);
326 lgraph.add_element(ly);
327
328 lgraph.add_raw(group_element::finish_group());
329 }
330
331 // Base typo for tic labels.
332 // NB: Assume pointsx/pointsy are monotonically increasing.
333 anntypo._M_size = ticsz;
334 if (gstate.is_visible(select::ticks))
335 {
336 lgraph.add_raw(group_element::start_group("ticks-" + gstate.title));
337
338 // X tic labels
339 auto mmx = minmax_element(pointsx.begin(), pointsx.end());
340 auto minx = *mmx.first;
341 auto maxx = *mmx.second;
342
343 const double xrange(maxx - minx);
344 const double xscale(gwidth / xrange);
345
346 const double ygo = gstate.marginy + gheight + asz;
347
348 // Filter tic labels to unique smallest subset of significant labels.
349 const double xticsteps = 100; // number of x tic labels
350 auto xnarrowl = [xticsteps](const double& d)
351 { return static_cast<uint>(d / xticsteps); };
352 vector<uint> xpointsi(pointsx.size());
353 std::transform(pointsx.begin(), pointsx.end(), xpointsi.begin(),
354 xnarrowl);
355 std::set<double> xpoints(xpointsi.begin(), xpointsi.end());
356
357 // Add x tic labels.
358 for (const double& x: xpoints)
359 {
360 const uint xui = x * xticsteps;
361 const double xto = gstate.marginx + (xui * xscale);
362 const string sxui = std::to_string(xui) + gstate.xticku;
363 styled_text(lgraph, sxui, {xto, ygo}, anntypo);
364 }
365
366 // Y tic labels
367 auto mmy = minmax_element(pointsy.begin(), pointsy.end());
368 auto miny = *mmy.first;
369 auto maxy = *mmy.second;
370
371 const double yrange(maxy - miny);
372 const double yscale(gheight / yrange);
373
374 const double xgo = gstate.marginx - asz;
375
376 // Filter tic labels to unique smallest subset of significant
377 // lables of yticsteps.
378 const double yticsteps = 10; // number of y tic labels
379 const double ydelta = yrange / yticsteps;
380
381 auto ynarrowl = [ydelta](const double& d)
382 { return static_cast<uint>(d / ydelta); };
383 vector<uint> ypointsi(pointsy.size());
384 std::transform(pointsy.begin(), pointsy.end(), ypointsi.begin(),
385 ynarrowl);
386 std::set<double> ypoints(ypointsi.begin(), ypointsi.end());
387
388 // Add y tic labels.
389 for (const double& y: ypoints)
390 {
391 const uint yui = y * ydelta;
392 const double yto = chartyo - (yui * yscale);
393 const string syui = std::to_string(yui) + gstate.yticku;
394 styled_text(lgraph, syui, {xgo, yto}, anntypo);
395 }
396
397 lgraph.add_raw(group_element::finish_group());
398 }
399}

402/// Returns a svg_element with the chart and labels
403/// Assume:
404/// vgrange x axis is monotonically increasing
405svg_element
406make_line_graph(const svg::area<> aplate, const vrange& points,
407 const graph_rstate& gstate,
408 const point_2t xrange = { },
409 const point_2t yrange = { })
410{
411 using namespace std;
412
413 // Scale vrange input to graph.
414 svg_element lgraph(gstate.title, "line graph", aplate, false);
415
416 // Split values and compute ranges for x/y axis.
417
418 // x
419 vector<double> pointsx(points.size());
420 std::transform(points.begin(), points.end(), pointsx.begin(),
421 [](const point_2t& pt) { return std::get<0>(pt); });
422 double minx = get<0>(xrange);
423 double maxx = get<1>(xrange);
424 if (minx == 0 && maxx == 0)
425 {
426 auto mmx = minmax_element(pointsx.begin(), pointsx.end());
427 minx = *mmx.first;
428 maxx = *mmx.second;
429 }
430
431 // y
432 vector<double> pointsy(points.size());
433 std::transform(points.begin(), points.end(), pointsy.begin(),
434 [](const point_2t& pt) { return std::get<1>(pt); });
435 double miny = get<0>(yrange);
436 double maxy = get<1>(yrange);
437 if (miny == 0 && maxy == 0)
438 {
439 auto mmy = minmax_element(pointsy.begin(), pointsy.end());
440 miny = *mmy.first;
441 maxy = *mmy.second;
442 }
443
444 // Locate graph area on plate area.
445 // aplate is total plate area with margins, aka
446 // pwidth = marginx + gwidth + marginx
447 // pheight = marginy + gheight + marginy
448 auto [ pwidth, pheight ] = aplate;
449 double gwidth = pwidth - (2 * gstate.marginx);
450 double gheight = pheight - (2 * gstate.marginy);
451 const double chartyo = pheight - gstate.marginy;
452 //const double chartxe = pwidth - gstate.marginx;
453
454 // Transform data points to scaled cartasian points in graph area.
455 vrange cpoints;
456 for (uint i = 0; i < points.size(); i++)
457 {
458 const point_2t& pt = points[i];
459 auto [ vx, vy ] = pt;
460
461 // At bottom of graph.
462 const double xlen = scale_value_on_range(vx, minx, maxx, 0, gwidth);
463 double x = gstate.marginx + xlen;
464
465 // Y axis grows up from chartyo.
466 const double ylen = scale_value_on_range(vy, miny, maxy, 0, gheight);
467 double y = chartyo - ylen;
468
469 cpoints.push_back(make_tuple(x, y));
470 }
471
472 // Add annotations first: any labels, metadata, ticmarks, legends
473 make_line_graph_annotations(aplate, pointsx, pointsy, gstate, lgraph);
474
475 // Plot path of points on cartesian plane.
476 // Grouped tooltips have to be the last, aka top layer of SVG to work (?).
477 //constexpr ushort line_strategy = line_1_polyline;
478 constexpr ushort line_strategy = line_2_polyline_tooltips;
479 if (gstate.is_visible(select::vector))
480 {
481 if constexpr(line_strategy == line_1_polyline)
482 {
483 // Use polylines and markerspoints
484 polyline_element pl1 = make_polyline(cpoints, gstate.lstyle,
485 gstate.sstyle);
486 lgraph.add_element(pl1);
487 }
488 if constexpr(line_strategy == line_2_polyline_tooltips)
489 {
490 // Use polyline base and set of marker paths with orignal values
491 // as tooltips on top.
492 lgraph.add_raw(group_element::start_group("polyline-" + gstate.title));
493 polyline_element pl1 = make_polyline(cpoints, gstate.lstyle,
494 gstate.sstyle);
495 lgraph.add_element(pl1);
496 lgraph.add_raw(group_element::finish_group());
497
498 lgraph.add_raw(group_element::start_group("values-" + gstate.title));
499 string markers = make_line_graph_markers_tips(points, cpoints,
500 gstate, 4);
501 lgraph.add_raw(markers);
502 lgraph.add_raw(group_element::finish_group());
503 }
504 }
505
506 return lgraph;
507}
