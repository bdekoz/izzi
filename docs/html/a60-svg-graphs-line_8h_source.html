88/// Return set of paths corresponding to marker shapes with tooltips.
89string
90make_line_graph_markers_tips(const vrange& points, const vrange& cpoints,
91 const graph_rstate& gstate, const double radius)
92{
93 string ret;
94 for (uint i = 0; i < points.size(); i++)
95 {
96 auto [ vx, vy ] = points[i];
97 auto [ cx, cy ] = cpoints[i];
98
99 // Generate displayed tooltip text....
100 string tipstr(gstate.title);
101 tipstr += k::newline;
102 tipstr += std::to_string(static_cast<uint>(vy));
103 tipstr += '%';
104 tipstr += k::comma;
105 tipstr += k::space;
106 tipstr += std::to_string(static_cast<uint>(vx));
107 tipstr += "ms";
108
109 const string& linecap = gstate.sstyle.linecap;
110 const bool roundp = linecap == "round" || linecap == "circle";
111 const bool squarep = linecap == "square";
112 const bool trianglep = linecap == "triangle";
113
114 // Markers default to closed paths that are filled with no stroke.
115 // Setting visible to vector | echo induces outline behavior.
116 style styl = gstate.lstyle;
117 styl._M_fill_opacity = 1;
118 if (gstate.is_visible(select::echo))
119 styl._M_fill_color = color::white;
120
121
122 // Circle Centered.
123 // svg::circle_element c = make_circle(cpoints[i], gstate.lstyle, r);
124 if (roundp)
125 {
126 circle_element c;
127 circle_element::data dc = { cx, cy, radius };
128 c.start_element();
129 c.add_data(dc);
130 c.add_style(styl);
131 c.add_raw(element_base::finish_hard);
132 c.add_title(tipstr);
133 c.add_raw(string { circle_element::pair_finish_tag } + k::newline);
134 ret += c.str();
135 }
136
137 // Square Centered.
138 // svg::rect_element r = (cpoints[i], gstate.lstyle, {2 * r, 2 * r});
139 if (squarep)
140 {
141 rect_element r;
142 rect_element::data dr = { cx - radius, cy - radius,
143 2 * radius, 2 * radius };
144 r.start_element();
145 r.add_data(dr);
146 r.add_style(styl);
147 r.add_raw(element_base::finish_hard);
148 r.add_title(tipstr);
149 r.add_raw(string { rect_element::pair_finish_tag } + k::newline);
150 ret += r.str();
151 }
152
153 // Triangle Centered.
154 // svg::path_element t = (cpoints[i], gstate.lstyle, {2 * r, 2 * r});
155 if (trianglep)
156 {
157 path_element p = make_path_triangle(cpoints[i], styl, radius, 120,
158 false);
159 p.add_title(tipstr);
160 p.add_raw(string { path_element::pair_finish_tag } + k::newline);
161 ret += p.str();
162 }
163
164 // Throw if marker style not supported.
165 if (!roundp && !squarep && !trianglep)
166 {
167 string m("make_line_graph_markers_tips:: ");
168 m += "linecap value invalid or missing, currently set to: ";
169 m += linecap;
170 m += k::newline;
171 throw std::runtime_error(m);
172 }
173 }
174 return ret;
175}

178/// Axis Labels
179/// Axis X/Y Ticmarks
180/// X line increments
181svg_element
182make_line_graph_annotations(const area<> aplate,
183 const vrange& points,
184 const graph_rstate& gstate,
185 const double xscalein = 1, const double yscalein = 1,
186 const typography typo = k::apercu_typo)
187{
188 using namespace std;
189 svg_element lanno(gstate.title, "line graph annotation", aplate, false);
190
191 // Locate graph area on plate area.
192 auto [ pwidth, pheight ] = aplate;
193 double gwidth = pwidth - (2 * gstate.marginx);
194 double gheight = pheight - (2 * gstate.marginy);
195 const double chartyo = pheight - gstate.marginy;
196 const double chartxo = gstate.marginx;
197 const double chartxe = pwidth - gstate.marginx;
198
199 // Base typo for annotations.
200 typography anntypo = typo;
201 anntypo._M_style = k::wcaglg_style;
202 anntypo._M_size = graph_rstate::th1sz;
203
204 // Axes and Labels
205 if (gstate.is_visible(select::axis))
206 {
207 lanno.add_raw(group_element::start_group("axes-" + gstate.title));
208
209 // Add axis labels.
210 point_2t xlabelp = make_tuple(pwidth / 2, chartyo + (gstate.marginy / 2));
211 styled_text(lanno, gstate.xlabel, xlabelp, anntypo);
212
213 point_2t ylabelp = make_tuple(chartxo / 2, pheight / 2);
214 styled_text(lanno, gstate.ylabel, ylabelp, anntypo);
215
216 // Add axis lines.
217 line_element lx = make_line({chartxo, chartyo}, {chartxe, chartyo},
218 gstate.lstyle);
219 line_element ly = make_line({chartxo, chartyo}, {chartxo, gstate.marginy},
220 gstate.lstyle);
221 lanno.add_element(lx);
222 lanno.add_element(ly);
223
224 lanno.add_raw(group_element::finish_group());
225 }
226
227 // Base typo for tic labels.
228 // NB: Assume pointsx/pointsy are monotonically increasing.
229 anntypo._M_size = graph_rstate::tticsz;
230
231 // Separate tic label values for each (x, y) axis, find ranges for each.
232 auto [ maxx, maxy ] = max_vrange(points, gstate.xticdigits, xscalein, yscalein);
233 auto minx = 0;
234 auto miny = 0;
235
236 const double xrange(maxx - minx);
237 const double xscale(gwidth / xrange);
238 const double yrange(maxy - miny);
239 const double yscale(gheight / yrange);
240
241 // const double xdelta = 1.0 / (10 * gstate.yticdigits);
242 const double xdelta = 0.1;
243 const double ydelta = yrange / gstate.yticdigits;
244
245 // Generate tic marks
246 if (gstate.is_visible(select::ticks))
247 {
248 // X tic labels
249 lanno.add_raw(group_element::start_group("tic-x-" + gstate.title));
250 const double ygo = gstate.marginy + gheight + graph_rstate::th1sz;
251 for (double x = minx; x < maxx + xdelta; x += xdelta)
252 {
253 const double xto = chartxo + (x * xscale);
254 ostringstream oss;
255 oss << fixed << setprecision(gstate.xticdigits) << x;
256 const string sxui = oss.str() + gstate.xticu;
257 styled_text(lanno, sxui, {xto, ygo}, anntypo);
258 }
259 lanno.add_raw(group_element::finish_group());
260
261 // Y tic labels
262 // Positions for left and right y-axis tic labels.
263 lanno.add_raw(group_element::start_group("tic-y-" + gstate.title));
264 const double xgol = gstate.marginx - graph_rstate::th1sz; // left
265 const double xgor = gstate.marginx + gwidth + graph_rstate::th1sz; // right
266 for (double y = miny; y < maxy + ydelta; y += ydelta)
267 {
268 const double yto = chartyo - (y * yscale);
269 const string syui = std::to_string(static_cast<uint>(y)) + gstate.yticu;
270 styled_text(lanno, syui, {xgol, yto}, anntypo);
271 styled_text(lanno, syui, {xgor, yto}, anntypo);
272 }
273 lanno.add_raw(group_element::finish_group());
274 }
275
276 // Horizontal lines linking y-axis tic labels, with magnification-ready text.
277 if (gstate.is_visible(select::linex))
278 {
279 lanno.add_raw(group_element::start_group("tic-y-lines-" + gstate.title));
280
281 style hlstyl = gstate.lstyle;
282 hlstyl._M_stroke_color = color::gray10;
283 for (double y = miny + ydelta; y < maxy + ydelta; y += ydelta)
284 {
285 const double yto = chartyo - (y * yscale);
286 line_element lxe = make_line({chartxo + graph_rstate::th1sz, yto}, {chartxe - graph_rstate::th1sz, yto}, hlstyl);
287 lanno.add_element(lxe);
288 }
289
290 lanno.add_raw(group_element::finish_group());
291 }
292
293 return lanno;
294}

297/// Returns a svg_element with the chart and labels
298/// Assume:
299/// vgrange x axis is monotonically increasing
300svg_element
301make_line_graph(const svg::area<> aplate, const vrange& points,
302 const graph_rstate& gstate,
303 const point_2t xrange, const point_2t yrange)
304{
305 using namespace std;
306 svg_element lgraph(gstate.title, "line graph", aplate, false);
307
308 auto [ minx, maxx ] = xrange;
309 auto [ miny, maxy ] = yrange;
310
311 // Locate graph area on plate area.
312 // aplate is total plate area with margins, aka
313 // pwidth = marginx + gwidth + marginx
314 // pheight = marginy + gheight + marginy
315 auto [ pwidth, pheight ] = aplate;
316 double gwidth = pwidth - (2 * gstate.marginx);
317 double gheight = pheight - (2 * gstate.marginy);
318 const double chartyo = pheight - gstate.marginy;
319
320 // Transform data points to scaled cartasian points in graph area.
321 vrange cpoints;
322 for (uint i = 0; i < points.size(); i++)
323 {
324 const point_2t& pt = points[i];
325 auto [ vx, vy ] = pt;
326
327 // At bottom of graph.
328 const double xlen = scale_value_on_range(vx, minx, maxx, 0, gwidth);
329 double x = gstate.marginx + xlen;
330
331 // Y axis grows up from chartyo.
332 const double ylen = scale_value_on_range(vy, miny, maxy, 0, gheight);
333 double y = chartyo - ylen;
334
335 cpoints.push_back(make_tuple(x, y));
336 }
337
338 // Plot path of points on cartesian plane.
339 // Grouped tooltips have to be the last, aka top layer of SVG to work (?).
340 //constexpr ushort line_strategy = line_1_polyline;
341 constexpr ushort line_strategy = line_2_polyline_tooltips;
342 if (gstate.is_visible(select::vector))
343 {
344 if constexpr(line_strategy == line_1_polyline)
345 {
346 // Use polylines and markerspoints
347 polyline_element pl1 = make_polyline(cpoints, gstate.lstyle,
348 gstate.sstyle);
349 lgraph.add_element(pl1);
350 }
351 if constexpr(line_strategy == line_2_polyline_tooltips)
352 {
353 // Use polyline base and set of marker paths with orignal values
354 // as tooltips on top.
355 lgraph.add_raw(group_element::start_group("polyline-" + gstate.title));
356 polyline_element pl1 = make_polyline(cpoints, gstate.lstyle,
357 gstate.sstyle);
358 lgraph.add_element(pl1);
359 lgraph.add_raw(group_element::finish_group());
360
361 lgraph.add_raw(group_element::start_group("values-" + gstate.title));
362 string markers = make_line_graph_markers_tips(points, cpoints,
363 gstate, 3);
364 lgraph.add_raw(markers);
365 lgraph.add_raw(group_element::finish_group());
366 }
367 }
368
369 return lgraph;
370}
