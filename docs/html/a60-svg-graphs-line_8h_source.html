72/// Base function for generating SVG markers in a defs section.
73marker_element
74make_marker_element(const std::string id,
75 const svg::area<> ma, const svg::point_2t mcp,
76 const std::string raw)
77{
78 svg::marker_element mkr;
79 mkr.start_element(id, ma, mcp);
80 mkr.add_raw(raw.c_str());
81 mkr.finish_element();
82 return mkr;
83}

122/// Create a set of markers bounded by a rectangle of size n.
123string
124make_marker_set_n(const double i)
125{
126 using svg::color::red;
127 using svg::color::wcag_lgray;
128 using svg::color::wcag_gray;
129 using svg::color::wcag_dgray;
130 using svg::k::b_style;
131 const svg::style style_r = { red, 1.0, red, 0.0, 0.5 };
132 const svg::style style_wcaglg = { wcag_lgray, 1.0, wcag_lgray, 0.0, 0.5 };
133 const svg::style style_wcagg = { wcag_gray, 1.0, wcag_gray, 0.0, 0.5 };
134 const svg::style style_wcagdg = { wcag_dgray, 1.0, wcag_dgray, 0.0, 0.5 };
135
136 // 4 / 2, etc.
137 const double h(i/2);
138 const string si = std::to_string(static_cast<uint>(i));
139 auto m1 = make_marker_circle("c" + si + "red", {i, i}, {h, h}, h, style_r);
140 auto m2 = make_marker_circle("c" + si + "wcaglg", {i, i}, {h, h}, h, style_wcaglg);
141 auto m3 = make_marker_circle("c" + si + "wcagdg", {i, i}, {h, h}, h, style_wcagdg);
142 auto m4 = make_marker_circle("c" + si + "black", {i, i}, {h, h}, h, b_style);
143 auto m5 = make_marker_triangle("t" + si + "black", {i, i}, {h, h}, h, b_style);
144 auto m6 = make_marker_triangle("t" + si + "wcagg", {i, i}, {h, h}, h, style_wcagg);
145 auto m7 = make_marker_x("x" + si + "wcagg", {i, i}, {h, h}, h, style_wcaglg);
146 auto m8 = make_marker_rect("r" + si + "wcaglg", {i, i}, {h, h}, style_wcaglg);
147 auto m9 = make_marker_rect("r" + si + "wcagdg", {i, i}, {h, h}, style_wcagdg);
148
149 std::ostringstream oss;
150 oss << m1.str() << m2.str() << m3.str() << m4.str()
151 << m5.str() << m6.str() << m7.str() << m8.str() << m9.str();
152 return oss.str();
153}

158make_markers(svg::svg_element& obj)
159{
160
161 svg::defs_element def;

170 def.finish_element();
171 obj.add_element(def);
172};

194/// Returns a svg_element with the chart and labels
195/// Assume:
196/// vgrange x axis is monotonically increasing
197svg_element
198make_line_graph(const svg::area<> aplate, const vrange& points,
199 const graph_state& gstate, const bool annotationsp = true)
200{
201 using namespace std;
202
203 // Scale vrange input to graph.
204 svg_element lgraph(gstate.title, "line graph", aplate, false);
205
206 // Split values and compute ranges for x/y axis.
207 vector<double> pointsy(points.size());
208 std::transform(points.begin(), points.end(), pointsy.begin(),
209 [](const point_2t& pt) { return std::get<1>(pt); });
210 auto mmy = minmax_element(pointsy.begin(), pointsy.end());
211 auto miny = *mmy.first;
212 auto maxy = *mmy.second;
213
214 // Locate graph area on plate area.
215 //
216 // aplate is plate area with margins, but graphable area without margins...
217 // pwidth = marginx + gwidth + marginx
218 // pheight = marginy + gheight + marginy
219 auto [ pwidth, pheight ] = aplate;
220 //double gwidth = pwidth - (2 * marginx);
221 double gheight = pheight - (2 * gstate.marginy);
222
223 // Transform data points to scaled cartasian points in graph area.
224 const double chartyo = pheight - gstate.marginy;
225 vrange cpoints;
226 uint i = 0;
227 for (const point_2t& pt : points)
228 {
229 auto [ vx, vy ] = pt;
230
231 // At bottom of graph.
232 point_2t xmatrix = to_point_in_1xn_matrix(aplate, points.size(),
233 i++, gstate.marginx, chartyo);
234 double x = get<0>(xmatrix);
235
236 // Y axis grows up from chartyo.
237 const double ylen = scale_value_on_range(vy, miny, maxy, 0, gheight);
238 double y = chartyo - ylen;
239
240 cpoints.push_back(make_tuple(x, y));
241 }
242
243 // Plot transformed points.
244 polyline_element pl1 = make_polyline(cpoints, gstate.lstyle,
245 gstate.dasharray, gstate.markerspoints);
246 lgraph.add_element(pl1);
247
248 if (annotationsp)
249 {
250 // Add axis labels.
251 point_2t xlabelp = make_tuple(pwidth / 2, chartyo + (gstate.marginy / 2));
252 styled_text(lgraph, gstate.xlabel, xlabelp, k::apercu_typo);
253
254 point_2t ylabelp = make_tuple(gstate.marginx / 2, pheight / 2);
255 styled_text_r(lgraph, gstate.ylabel, ylabelp, k::apercu_typo, 90);
256 }
257
258 return lgraph;
259}
