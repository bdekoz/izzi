40/// Base function for generating SVG markers in a defs section.
41marker_element
42make_marker_element(const std::string id,
43 const svg::area<> ma, const svg::point_2t mcp,
44 const std::string raw)
45{
46 svg::marker_element mkr;
47 mkr.start_element(id, ma, mcp);
48 mkr.add_raw(raw.c_str());
49 mkr.finish_element();
50 return mkr;
51}

90/// Make black/white/wcag markers.
91void
92make_markers(svg::svg_element& obj)
93{
94 using svg::color::red;
95 using svg::color::wcag_lgray;
96 using svg::color::wcag_gray;
97 using svg::color::wcag_dgray;
98 const svg::style style_r = { red, 1.0, red, 0.0, 0.5 };
99 const svg::style style_wcaglg = { wcag_lgray, 1.0, wcag_lgray, 0.0, 0.5 };
100 const svg::style style_wcagg = { wcag_gray, 1.0, wcag_gray, 0.0, 0.5 };
101 const svg::style style_wcagdg = { wcag_dgray, 1.0, wcag_dgray, 0.0, 0.5 };
102
103 obj.add_element(make_marker_circle("c4red", {4, 4}, {2, 2}, 2, style_r));
104
105 obj.add_element(make_marker_circle("c4wcaglg", {4, 4}, {2, 2}, 2, style_wcaglg));
106 obj.add_element(make_marker_circle("c4wcagdg", {4, 4}, {2, 2}, 2, style_wcagdg));
107 obj.add_element(make_marker_circle("c4black", {4, 4}, {2, 2}, 2, svg::k::b_style));
108
109 obj.add_element(make_marker_triangle("t4black", {4, 4}, {2, 2}, 2, svg::k::b_style));
110 obj.add_element(make_marker_triangle("t4wcagg", {4, 4}, {2, 2}, 2, style_wcagg));
111
112 obj.add_element(make_marker_x("x4wcagg", {4, 4}, {2, 2}, 2, style_wcaglg));
113
114 obj.add_element(make_marker_rect("r4wcaglg", {4, 4}, {2, 2}, style_wcaglg));
115 obj.add_element(make_marker_rect("r4wcagdg", {4, 4}, {2, 2}, style_wcagdg));
116};
