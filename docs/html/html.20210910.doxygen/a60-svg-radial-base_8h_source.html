44 /// Given rdenom scaling factor and SVG canvas, compute max effective
45 /// radius value if centered on the page/frame.

50  return leastside / rdenom;

54 /**
55  Max number of non-overlapping degrees in radial form,
56  as a tuple of (min, max).
57 
58  Total degrees in a circle are 360, but then the beginning and the
59  end of the radial display are in the same place, which may
60  confusing. So, shave a bit off both ends for the optics, so that
61  there is a gap between the beginning and end of the generated
62  radial viz. The default is such that the beginning and the end have
63  a discernable gap.
64 */
65 point_2t&
66 get_radial_range()
67 {
68  static point_2t rrange = { 10, 350 };
69  return rrange;
70 }

169 /**
170  Circumference adjustment such that text/object of height distance
171  is centered on arc from origin to kusama high-orbit glyphs.
172 */
173 constexpr double
174 adjust_angle_at_orbit_for_distance(double r, double distance)
175 {
176  // Angle between sequential id's.
177  //
178  // Given r,
179  // point p on circumference r distance from origin with angle d
180  // point pprime on circumference r from origin with angle dprime
181  // distance is the cartesian distance between p and pprime
182  //
183  // then
184  // dprime = 2 * theta, where theta from sin(theta) = (rspace / 2) / r.
185  double angleprime = std::asin((distance / 2) / r);
186  double angleprimed = angleprime * (180 / k::pi);
187  return angleprimed;
188 }

235 /**
236  Easier to read radial text that switches orientation at 180
237  degrees, instead of going upside-down as it follows the circle
238  around the circumfrence, aka mirrored or symmetric radial text.
239 
240  Text with typography, arranged around an origin of a circle with radius r.
241  Text is align left CW (1,180)
242  Text is aligh right CCW (181, 359)
243 
244  NB: Assume value is un-adjusted, aka from get_angle
245 */
246 void
247 radial_text_r(svg_element& obj, const typography& typo, string text,
248  const int r, const point_2t origin, const double deg = 0.0)
249 {
250  const double angletextoffset = 0;
251  if (deg <= 180)
252  {
253  auto dcw = zero_angle_north_cw(deg);
254  dcw -= angletextoffset;
255  auto [ x, y ] = get_circumference_point_d(dcw, r, origin);
256  radial_text_cw(obj, typo, text, x, y, dcw);
257  }
258  else
259  {
260  // Mapping some values...
261 
262  // d 210 cw -> d 150 ccw
263  // d 240 cw -> d 120 ccw
264  // d 270 cw -> d 90 ccw
265  // d 330 cw -> d 30 ccw
266 
267  // 210 - 180 == 30 -> 180 - 30 == 150
268  // 240 - 180 == 60 -> 180 - 60 == 120
269  // 330 - 180 == 150 -> 180 - 150 == 30
270 
271  auto dp = deg - 180;
272  auto dccw = zero_angle_north_ccw(180 - dp);
273  dccw += angletextoffset;
274  auto [ x, y ] = get_circumference_point_d(dccw, r, origin);
275  radial_text_ccw(obj, typo, text, x, y, dccw);
276  }
277 }

280 // Radiate clockwise from 0 to 35x degrees about origin, placing each
281 // id at a point on the circumference. Duplicate ids splay, stack,
282 // or append/concatencate at, after, or around that point.
283 /// Spread ids on either side of an origin point, along circumference
284 /// path.
285 void
286 splay_ids_around(svg_element& obj, const typography& typo,
287  const strings& ids, const double angled,
288  const point_2t origin, double r, double rspace,
289  const bool satellitep = false)
290 {
291  // If rspace is set for labels, and isn't adjusted for this radius,
292  // make sure it is set low.
293  if (rspace > r * 2)
294  rspace = r * 2;
295 
296  const double angleprimed = adjust_angle_at_orbit_for_distance(r, rspace);
297  const double maxdeg = angleprimed * (ids.size() - 1);
298  double angled2 = angled - (maxdeg / 2);
299 
300  // NB: value label is at r + rspace, so rprime has to be longer.
301  double rlabel = r + rspace;
302  double rprime = rlabel + (rspace * 2);
303  for (const string& s: ids)
304  {
305  if (satellitep)
306  {
307  // Make a ring on point on the kusama circle for the id.
308  auto [ x, y ] = get_circumference_point_d(angled2, r, origin);
309  auto rring = rspace / 2;
310  const id_render_state ridst = get_id_render_state(s);
311  style rstyl = ridst.styl;
312  point_2d_to_circle(obj, x, y, rstyl, rring);
313  }
314 
315  radial_text_r(obj, typo, s, rprime, origin, angled2);
316  angled2 += angleprimed;
317  }
318 }
