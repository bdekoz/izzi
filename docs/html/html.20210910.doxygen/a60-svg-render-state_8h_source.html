118 /// Settings for rendering collection-derived objects.

128  bool weigh;
129 
130  // Genderate colors from color bands (true) or use finite colors (false).

133  // Alternate output naming mode via scheme (true = info, false = map)

136  render_state_base(const double o = 0.10,

229 /// Add value to cache with base style of styl, colors klr, visibility vis.
230 void
231 add_to_id_render_state_cache(const string id, const style styl,
232  const k::select vis)
233 {
234  id_render_state_umap& cache = get_id_render_state_cache();
235 
236  id_render_state state(styl, id);
237  state.set(state.visible_mode, vis);
238  cache.insert(std::make_pair(id, state));

242 /// Given identifier/name/id, get corresponding id_render_state from cache.
243 const id_render_state
244 get_id_render_state(const string id)
245 {
246  const id_render_state_umap& cache = get_id_render_state_cache();
247 
248  id_render_state ret;
249  if (cache.count(id) == 1)
250  {
251  auto iter = cache.find(id);
252  ret = iter->second;
253  }
254  else
255  {
256  // If default string is set in the cache, use it.
257  if (cache.count("") == 1)
258  {
259  auto iter = cache.find("");
260  ret = iter->second;
261  }
262  }
263  return ret;
264 }

267 /// Roll through render states given in values squentially,
268 /// index starts with zero.
269 const id_render_state
270 traverse_states(const strings& values)
271 {
272  static uint indx(0);
273  string value;
274  if (indx < values.size())
275  {
276  value = values[indx];
277  ++indx;
278  }
279  else
280  {
281  // Start over.
282  indx = 0;
283  value = values[indx];
284  }
285  return get_id_render_state(value);
286 }
