354  const std::string
355  add_attribute(const svg::unit utype = svg::unit::pixel) const
356  {
357  const string name("__name");
358  const string size("__size");
359  const string anchor("__anchor");
360  const string align("__align");
361  const string len("__lentype");
362 
363  std::string strip1 = \
364  R"(font-family="__name" font-size="__size__lentype" text-anchor="__anchor" text-align="__align" )";
365 
366  string_replace(strip1, name, _M_face);
367  string_replace(strip1, size, std::to_string(_M_size));
368  string_replace(strip1, anchor, to_string(_M_anchor));
369  string_replace(strip1, align, to_string(_M_align));
370  string_replace(strip1, len, svg::to_string(utype));
371 
372  const std::string weight("__weight");
373  const std::string property("__prop");
374 
375  std::string strip2 = \
376  R"(font-weight="__weight" font-style="__prop")";
377 
378  string_replace(strip2, weight, to_string(_M_w));
379  string_replace(strip2, property, to_string(_M_p));
380 
381  // Add dominant baseline only if necessary.
382  string stripb;
383  if (_M_baseline != baseline::none)
384  {
385  stripb += "dominant-baseline=";
386  stripb += k::quote;
387  stripb += to_string(_M_baseline);
388  stripb += k::quote;
389  stripb += k::space;
390  }
391  return strip1 + stripb + strip2;
392  }
