416 /// Color quantified as components in aggregate type with RGB values.
417 /// aka like Scalar in OpenCV.
418 struct colorq
419 {
420  using itype = unsigned short;
421 
422  itype r;
423  itype g;
424  itype b;
425 
426  // Return "rgb(64, 64, 64)";
427 
428  static string
429  to_string(colorq s)
430  {
431  std::ostringstream oss;
432  oss << "rgb(" << s.r << ',' << s.g << ',' << s.b << ")";
433  return oss.str();
434  }
435 
436  // From "rgb(64, 64, 64)";
437  static colorq
438  from_string(string s)
439  {
440  // Kill rgb() enclosing, if be.
441  if (s.empty() || s.size() < 5 || s[0] != 'r')
442  {
443  string m("colorq::from_string input is not in rbg form: ");
444  m += s;
445  m += k::newline;
446  throw std::runtime_error(m);
447  }
448  else
449  {
450  s.pop_back();
451  s = s.substr(4);
452  }
453 
454  // String stream which eats whitespace and knows number separation.
455  std::istringstream iss(s);
456  iss >> std::skipws;
457 
458  char c(0);
459 
460  ushort rs(0);
461  iss >> rs;
462  itype r = static_cast<itype>(rs);
463  iss >> c;
464 
465  ushort gs(0);
466  iss >> gs;
467  itype g = static_cast<itype>(gs);
468  iss >> c;
469 
470  ushort bs(0);
471  iss >> bs;
472  itype b = static_cast<itype>(bs);
473 
474  return colorq(r, g, b);
475  }
476 
477  colorq() = default;
478  colorq(const colorq&) = default;
479  colorq& operator=(const colorq&) = default;
480 
481  // auto operator<=>(const colorq&) const = default;
482 
483  colorq(itype ra, itype ga, itype ba) : r(ra), g(ga), b(ba) { }
484 
485  colorq(const color e)
486  {
487  colorq klr = from_string(svg::to_string(e));
488  r = klr.r;
489  b = klr.b;
490  g = klr.g;
491  }
492 };

502 /**
503  Combine color a with color b in percentages ad and ab, respectively.
504 
505  To average, constrain paramters ad and ab such that: ad + ab == 2.
506 
507  Like so:
508  ushort ur = (a.r + b.r) / 2;
509  ushort ug = (a.g + b.g) / 2;
510  ushort ub = (a.b + b.b) / 2;
511 */
512 colorq
513 combine_two_colorq(const colorq& a, const double ad,
514  const colorq& b, const double bd)
515 {
516  double denom = ad + bd;
517  double ur = ((a.r * ad) + (b.r * bd)) / denom;
518  double ug = ((a.g * ad) + (b.g * bd)) / denom;
519  double ub = ((a.b * ad) + (b.b * bd)) / denom;
520 
521  using itype = colorq::itype;
522  itype cr = static_cast<itype>(ur);
523  itype cg = static_cast<itype>(ug);
524  itype cb = static_cast<itype>(ub);
525  return colorq { cr, cg, cb };
526 }
527 
528 /// Average two colors, return the result.
529 colorq
530 average_two_colorq(const colorq& a, const colorq& b)
531 { return combine_two_colorq(a, 1.0, b, 1.0); }

685 /**
686  Generate a color band from starting hue and seeds.
687 
688  Algorithm is average two known good, where two picked randomly.
689 
690  Return type is a vector of generated colorq types.
691 */
692 colorqs
693 make_color_band(const colorband& cb, const ushort neededh)
694 {
695  // Find starting hue and number of samples in the color band.
696  color c = std::get<0>(cb);
697  ushort hn = std::get<1>(cb);
698 
699  // Find initial offset.
700  auto itr = std::find(spectrum.begin(), spectrum.end(), c);
701  if (itr == spectrum.end())
702  {
703  string m("collection::make_color_band: " + to_string(c));
704  throw std::runtime_error(m);
705  }
706 
707  // Randomness.
708  static std::mt19937_64 rg(std::random_device{}());
709 
710  // Setup random picker of sample hues in band.
711  auto disti = std::uniform_int_distribution<>(0, hn - 1);
712 
713  // Set up random percentage for combining two colors.
714  auto distr = std::uniform_real_distribution<>(0, 1);
715 
716  // Generate new from averaging random samples, cache in return vector.
717  colorqs cband;
718  for (ushort i = 0; i < neededh; ++i)
719  {
720  // New color.
721  ushort o1 = disti(rg);
722  ushort o2 = disti(rg);
723  color c1 = *(itr + o1);
724  color c2 = *(itr + o2);
725 
726  // Combine.
727  double c1r = distr(rg);
728  double c2r = 2.0 - c1r;
729  colorq cgen = combine_two_colorq(c1, c1r, c2, c2r);
730  cband.push_back(cgen);
731  }
732 
733  return cband;
734 }
