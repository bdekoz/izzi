33  const double angled, const size_type v,

35  const typography& typo,

52  Draw glyph on ray from origin as part of kusama.

54 int
55 radiate_glyph(svg_element& obj, const point_2t origin, const double angled,
56  const id_render_state idst,
57  const int kr, const int rspace, const int rstart)
58 {
59  // Kusama circle radius.
60  // Assumed to scale per value/value_max ratio.
61  const double angleda = adjust_angle_rotation(angled, k::rrotation::cw);
62 
63  const string glyphtext = idst.name;
64  const double glyphscale = idst.multiple;
65  const double glyphrotate = idst.rotate;
66 
67  int glyphr(0);
68  if (idst.is_visible(svg::k::select::svg))
69  {
70  // SVG to be inserted is
71  // - square canvas width/height of 100 pixels
72  // - glyph centered in this canvas
73  // - glyph circle diameter = glyphscale value in pixels
74  // Implies:
75  // kr * 2 is desired width of circle in glyph.
76  const double scale(kr * 2 / glyphscale);
77  const double scaledsize = scale * 100;
78  const double scaledglyph = scale * glyphscale;
79  const double svgr = rstart + rspace + (scaledglyph / 2);
80  point_2t p = get_circumference_point_d(angleda, svgr, origin);
81  insert_svg_at(obj, glyphtext, p, 100, scaledsize,
82  angleda + glyphrotate);
83  glyphr += rspace + scaledsize;
84  }
85 
86  if (idst.is_visible(svg::k::select::vector))
87  {
88  const int vr = rstart + rspace + kr;
89  point_2t p = get_circumference_point_d(angleda, vr, origin);
90  auto [ x, y ] = p;
91  point_2d_to_circle(obj, x, y, idst.styl, kr);
92  glyphr += rspace + (2 * kr);
93  }
94 
95  return glyphr;
96 }

99 /**
100  Draw glyph and id on ray from origin as part of kusama.
101  skip (origin to rstart) + rspace + glyph of radius + rspace + id
102 
103  @origin is center of the primary/base circle for kusama renderings.
104  @radius is of the kusama satellite circle on the ray.
105  @rstart is the length from origin along the ray to begin rendering
106  @rspace is the distance between text/vector/glyph elements
107 
108  Returns the length along the arc of the generated glyph.
109 */
110 int
111 radiate_glyph_and_id(svg_element& obj, const point_2t origin,
112  const size_type v, const size_type value_max,
113  const int radius, const int rspace, const int rstart,
114  const string id, const typography& typo)
115 {
116  // Length used of glyphs along radiated ray from origin, if any.
117  int glyphr(0);
118  const double angled = get_angle(v, value_max);
119  const id_render_state idst = get_id_render_state(id);
120 
121  // Switch based on id_render_state settings.
122  if (idst.is_visible(svg::k::select::glyph))
123  {
124  const double kr((double(v) / value_max) * radius);
125  glyphr += radiate_glyph(obj, origin, angled, idst, kr, rspace, rstart);
126  }
127 
128  // Id name.
129  if (idst.is_visible(svg::k::select::text) && !id.empty())
130  {
131  const int idr = rstart + glyphr + rspace;
132  radial_text_r(obj, typo, id, idr, origin, angled);
133 
134  // NB: This is only an estimate of the text block size.
135  // Should be getComputedTextLength
136  glyphr += id.size() * char_width_to_px(typo._M_size);
137  }
138 
139  return glyphr;
140 }

143 /**
144  Radiate clockwise from 0 to 35x degrees about origin, placing each
145  id at a point cluster on the circumference. A point cluster is a
146  circle whos radius is proportionate to the number of duplicate ids
147  at that point. Duplicate ids splay, stack, or
148  append/concatencate at, after, or around that point cluster.
149 */
150 void
151 kusama_collision_transforms(const point_2t origin,
152  std::vector<size_type> vuvalues,
153  std::vector<point_2tn>& vpointns,
154  const size_type value_max, const int radius,
155  const int rspace, const bool outwardp = true)
156 {
157  // Threshold is the range such that a value is considered adjacent
158  // for collisions. If v > previous value + threshold, then the
159  // points are not considered neighbors.
160  const int threshold(200);
161 
162  // Massage data to fit without overlaps, given:
163  // 1 unique set of values
164  // 2 for each unique value, the ids that match
165  // 3 for each unique value, the proposed point and size of the circle
166  for (uint i = 0; i < vpointns.size(); ++i)
167  {
168  auto v = vuvalues[i];
169  auto& pn = vpointns[i];
170  auto& [ p, n ] = pn;
171  double rcur = rspace * n;
172 
173  // Fixed angle, just push point along ray from origin until no
174  // conflicts.
175  const double angledo = get_angle(v, value_max);
176  const double angled = adjust_angle_rotation(angledo, k::rrotation::cw);
177  double angler = (k::pi / 180.0) * angled;
178 
179  // Points near p that are under threshold (and lower-indexed in
180  // vpointns).
181  uint neighbors(0);
182 
183  // Find neighbors.
184  uint j(i);
185  while (i > 0 && j > 0)
186  {
187  j -= 1;
188 
189  // Only if within threshold, know values are ordered.
190  if (vuvalues[i] - vuvalues[j] < threshold)
191  ++neighbors;
192  else
193  break;
194  }
195  std::clog << i << k::tab << "neighbors: " << neighbors
196  << std::endl;
197 
198  // Search backward and fixup.... in practice results in overlap
199  // with lowest-index or highest-index neighbor.
200  //
201  // So... search forward and fixup. Not ideal; either this or
202  // move to collision detection with multiple neighbor points.
203  //
204  // If collisions, extend radius outward from origin and try
205  // again until the collision is removed and the previous
206  // neighbor circles don't overlap.
207  double rext = radius + rcur; // ??? XXX
208  for (uint k = neighbors; k > 0; --k)
209  {
210  // Get neighbor point, starting with lowest-index neighbor.
211  auto& prevpn = vpointns[i - k];
212  auto& [ prevp, prevn ] = prevpn;
213  double rprev = radius * prevn; // ??? XXX
214 
215  while (detect_collision(p, rcur, prevp, rprev))
216  {
217  // Find new point further out from origin.
218  if (outwardp)
219  rext += (2.5 * rcur);
220  else
221  rext -= (2.5 * rcur);
222  p = get_circumference_point(angler, rext, origin);
223  }
224  }
225  }
226 }

229 /**
230  Draw these ids around a kusama low orbit circle on arcs from the origin circle.
231 
232  Simplest version, make satellite circle on circumfrence and splay
233  or append id's around it.
234 */
235 int
236 kusama_ids_orbit_high(svg_element& obj, const strings& ids, const point_2t origin,
237  const size_type v, const size_type value_max,
238  const int radius, const int rspace, const typography& typo,
239  const bool wbyvaluep, const bool satellitep)
240 {
241  // Center of glyph, a point on origin circle circumference.
242  const double angled = get_angle(v, value_max);
243 
244  int glyphr(0);
245  auto linelen = rspace;
246  glyphr = radiate_line_and_value(obj, origin, angled, v,
247  rspace, radius, linelen, typo);
248 
249  // Add number of characters of value as string * size of each character.
250  glyphr += get_label_spaces(value_max) * char_width_to_px(typo._M_size);
251 
252  if (satellitep)
253  {
254  // If satellitep, draw circle in the default style in low orbit to
255  // hang the rest of the glyphs off of, in high orbit...
256  glyphr += radiate_glyph_and_id(obj, origin, v, value_max, radius, rspace,
257  radius + glyphr, "", typo);
258  }
259 
260  // Variation based on splay_ids_around center point,
261  // where center is point on arc from origin...
262  glyphr += rspace;
263  const double anglea = adjust_angle_at_orbit_for_distance(radius + glyphr, rspace);
264  const double maxdeg = anglea * (ids.size() - 1);
265  double angled2 = angled - (maxdeg / 2);
266 
267  int maxglyphr2(0);
268  for (const string& id : ids)
269  {
270  int glyphr2(0);
271  const id_render_state idst = get_id_render_state(id);
272  if (idst.is_visible(svg::k::select::glyph))
273  {
274  double kr(0);
275  if (wbyvaluep)
276  kr = ((double(v) / value_max) * radius);
277  else
278  kr = 4; // XXX configurable via argument? wbyvaluep?
279  glyphr2 = radiate_glyph(obj, origin, angled2, idst, kr, rspace, glyphr);
280  }
281 
282  if (idst.is_visible(svg::k::select::text) && !id.empty())
283  {
284  const int idr = glyphr + glyphr2 + rspace;
285  radial_text_r(obj, typo, id, idr, origin, angled2);
286 
287  // NB: This is only an estimate of the text block size.
288  // Should be getComputedTextLength
289  glyphr2 += id.size() * char_width_to_px(typo._M_size);
290  }
291 
292  // Advance and bookkeeping for the next round.
293  angled2 += anglea;
294  maxglyphr2 = std::max(maxglyphr2, glyphr2);
295  }
296  glyphr += maxglyphr2;
297 
298  return glyphr;
299 }

341 /**
342  Radiate as *_per_uvalue_on_arc function, but group similar
343  values such that they are globbed into a satellite circle, ids
344  splayed around the satellite, and not written on top of each other
345  on the same arc/angle.
346 
347  Framing circle radius is the result of dividing page dimensions by (rdenom).
348 
349  Satellite circle radius is the product of the number of ids with
350  the same value times a base multipler (rbase).
351 
352  When overlap is detected, move outward on radius if true, otherwise
353  move in.
354 */
355 svg_element
356 kusama_ids_per_uvalue_on_arc(svg_element& obj, const point_2t origin,
357  const typography& typo, const id_value_umap& ivm,
358  const size_type value_max, const int radius,
359  const int rspace, const bool weighbyvaluep = true,
360  const bool collisionp = false)
361 {
362  // Convert from string id-keys to int value-keys, plus an ordered
363  // set of all the unique values.
364  value_set uvalues;
365  value_id_ummap uvaluemm = to_value_id_mmap(ivm, uvalues);
366 
367  // Map out preliminary data points. For each unique value in vuvalues
368  //
369  // - VIDS
370  // - construct a vector of strings that have that value, sorted
371  // - short to long
372  //
373  // - VPOINTNS
374  // - constuct a vector of points on the circumference with weight
375  std::vector<size_type> vuvalues(uvalues.begin(), uvalues.end());
376  std::vector<strings> vids;
377  std::vector<point_2tn> vpointns;
378  for (const auto& v : vuvalues)
379  {
380  // Extract all the ids for a given value.
381  auto irange = uvaluemm.equal_range(v);
382  auto ibegin = irange.first;
383  auto iend = irange.second;
384  strings ids;
385  for (auto i = ibegin; i != iend; ++i)
386  ids.push_back(i->second);
387  sort_strings_by_size(ids);
388  vids.push_back(ids);
389 
390  // Find initial point on the circumference of the circle closest
391  // to current value, aka initial circumference point (ICP).
392  const double angledo = get_angle(v, value_max);
393  const double angled = adjust_angle_rotation(angledo, k::rrotation::cw);
394  point_2t p = get_circumference_point_d(angled, radius, origin);
395  vpointns.push_back(std::make_tuple(p, ids.size()));
396  }
397 
398  // Remove zero value ids.
399  if (vuvalues.front() == 0)
400  {
401  auto& ids = vids[0];
402  std::clog << "eliding ids with value zero: " << std::endl;
403  for (const auto& s: ids)
404  std::clog << s << std::endl;
405 
406  vuvalues.erase(vuvalues.begin());
407  vids.erase(vids.begin());
408  vpointns.erase(vpointns.begin());
409  }
410 
411 
412  // Further normalization and collision-avoidance.
413  if (collisionp)
414  kusama_collision_transforms(origin, vuvalues, vpointns, value_max,
415  radius, rspace);
416 
417 
418  // Draw resulting points, ids, values.
419  // NB: vpointns valued from smallest to largest, so reverse so that
420  // smaller is more visible.
421  constexpr bool overlayrayp = false;
422  for (uint i = 0; i < vpointns.size(); ++i)
423  {
424  int j = vpointns.size() - 1 - i;
425  auto& ids = vids[j];
426  auto v = vuvalues[j];
427 
428  // Draw this id's kusama circle on the circumference of origin
429  // circle.
430  kusama_ids_orbit_low(obj, ids, origin, v, value_max, radius, rspace,
431  typo, weighbyvaluep);
432 
433  // Iff overlay rays to check text and glyph alignment.
434  if (overlayrayp)
435  {
436  const style rstyl = { color::red, 1.0, color::red, 1.0, 2 };
437  const double angled = get_angle(v, value_max);
438  radiate_line_and_value(obj, origin, angled, v, radius, rspace,
439  radius * 4, typo, rstyl);
440  }
441  }
442 
443  return obj;
444 }
