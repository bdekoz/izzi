24 /**
25 Selected or Active in render area.
26 Make discrete element or layer (visible, outline, etc) if true.
27 Used as a (visibility, outline, etc.) bitmask
28 */
29 enum class select : uint
30 {
31 none = 1u << 0, ///< nothing
32 cartography = 1u << 1, ///< cartographic elements
33 vector = 1u << 2, ///< svg path, circle, rectangle, etc.
34 cloud = 1u << 3, ///< opacity color fill version of vector
35 blur = 1u << 4, ///< blur or gradient version of vector
36 echo = 1u << 5, ///< b & w outline version of vector
37 text = 1u << 6, ///< title, metadata, header
38 legend = 1u << 7, ///< text and symbol legends
39 exitnode = 1u << 8, ///< tor exit node infrastructure
40 telecom = 1u << 9, ///< telecom infrastructure
41 glyph = 1u << 10, ///< glyph
42 image = 1u << 11, ///< image
43 svg = 1u << 12, ///< svg element, perhaps nested
44 alt = 1u << 13, ///< alternate use specified in situ
45 background = 1u << 14, ///< background elements
46 raster = 1u << 15, ///< raster (pixel) elements
47 top = 1u << 16, ///< top
48 bottom = 1u << 17, ///< bottom
49 inner = 1u << 18, ///< inner
50 outer = 1u << 19, ///< outer
51 next = 1u << 20, ///< next
52 previous = 1u << 21, ///< previous
53 even = 1u << 22, ///< even
54 odd = 1u << 23, ///< odd
55 all = 1u << 24, ///< all elements and layers
56 _S_end = 1u << 25 ///< last
57 };

101 /**
102 Some high-level nobs for rendering: scale tuning.
103
104 small == defense distributed == 70k-300k
105 medium == westworld == 2-4M
106 large == stranger things == 10M+
107
108 [1-5]xsmall == reduction from least to greatest
109 1x == baseline
110 [1-5]xlarge == enlargement from least to greatest
111 */
112 enum class scale
113 {
114 r5s, r4s, r3s, r2s, r1s,
115 xxsmall, xsmall,
116 small,
117 medium, baseline,
118 large,
119 xlarge, xxlarge,
120 e1s, e2s , e3s, e4s, e5s,
121 };

157/// Settings for rendering collection-derived objects.
158struct render_state_base
159{
160 k::select visible_mode;
161 k::select outline_mode;
162 k::scale scale_mode;
163
164 double opacity;
165 int text_scale_max;
166
167 // Values with similar geo coordinates counted in as one meta coordinate.
168 bool weigh;
169
170 // Genderate colors from color bands (true) or use finite colors (false).
171 bool color_generated;
172
173 // Alternate output naming mode via scheme (true = info, false = map)
174 bool alt;
175
176 render_state_base(const double o = 0.10,
177 const k::scale rscale = k::scale::medium)
178 : visible_mode(k::select::none), outline_mode(k::select::none),
179 scale_mode(rscale), opacity(o), text_scale_max(0),
180 weigh(false), color_generated(true), alt(false)
181 { }
182
183 render_state_base(const render_state_base&) = default;
184
185 bool
186 is_visible(const k::select v) const
187 {
188 using __utype = typename std::underlying_type<k::select>::type;
189 __utype isv(static_cast<__utype>(visible_mode & v));
190 return isv > 0;
191 }
192
193 void
194 set(k::select& a, const k::select& b)
195 { a = b; }
196
197 void
198 clear(k::select& a, const k::select& b)
199 { a &= ~b; }
200
201 void
202 flip(k::select& a, const k::select& b)
203 { a |= b; }
204
205 string
206 mangle() const
207 {
208 using k::select;
209 string mangled;
210
211 bool foundp = false;
212
213 const uint vi = static_cast<uint>(visible_mode);
214 if (vi > 0)
215 {
216 mangled += "visible-mode-";
217 mangled += std::to_string(vi);
218 foundp = true;
219 }
220
221 const uint oi = static_cast<uint>(outline_mode);
222 if (oi > 0)
223 {
224 if (foundp)
225 mangled += k::hyphen;
226 else
227 foundp = true;
228 mangled += "outline-mode-";
229 mangled += std::to_string(oi);
230 }
231
232 if (opacity > 0)
233 {
234 if (foundp)
235 mangled += k::hyphen;
236 else
237 foundp = true;
238 mangled += "opacity-";
239 const double o(opacity * 100);
240 const uint oi = static_cast<uint>(o);
241 mangled += std::to_string(oi);
242 }
243
244 return mangled;
245 }
246};

254 colormap klrs;

257 get_color(const string& s) const
258 {
259 auto i = klrs.find(s);

279/// Named render state.
280/// Datum to take id string and tie it to visual representation.
281struct id_render_state: public render_state_base
282{
283 style styl; // Overrides render_state_base.opacity
284 string name;
285 double rotate; // Degrees to rotate to origin (CW), if any.
286 double multiple; // Scale factor, if any.
287
288 static style dstyl;
289
290 explicit
291 id_render_state(const style s = dstyl, const string f = "",
292 const double angle = 0.0, const double scale = 1.0)
293 : styl(s), name(f), rotate(angle), multiple(scale) { }
294
295 id_render_state(const id_render_state&) = default;
296 id_render_state& operator=(const id_render_state&) = default;
297};

312/// Add value to cache with base style of styl, colors klr, visibility vis.
313void
314add_to_id_render_state_cache(const string id, const style styl,
315 const k::select vis)
316{
317 id_render_state_umap& cache = get_id_render_state_cache();
318
319 id_render_state state(styl, id);
320 state.set(state.visible_mode, vis);
321 cache.insert(std::make_pair(id, state));
322}

325/// Given identifier/name/id, get corresponding id_render_state from cache.
326const id_render_state
327get_id_render_state(const string id)
328{
329 const id_render_state_umap& cache = get_id_render_state_cache();
330
331 id_render_state ret;
332 if (cache.count(id) == 1)
333 {
334 auto iter = cache.find(id);
335 ret = iter->second;
336 }
337 else
338 {
339 // If default string is set in the cache, use it.
340 if (cache.count("") == 1)
341 {
342 auto iter = cache.find("");
343 ret = iter->second;
344 }
345 }
346 return ret;
347}

350/// Roll through render states given in values squentially,
351/// index starts with zero.
352const id_render_state
353traverse_states(const strings& values)
354{
355 static uint indx(0);
356 string value;
357 if (indx < values.size())
358 {
359 value = values[indx];
360 ++indx;
361 }
362 else
363 {
364 // Start over.
365 indx = 0;
366 value = values[indx];
367 }
368 return get_id_render_state(value);
369}
