203/// Set and Get working spectrum, aka default palette.
204auto&
205active_spectrum()
206{
207 //static auto spectrum = izzi_palette;
208 //static auto spectrum = jp_palette;
209 static auto spectrum = ciecam02_palette;
210
211 static bool initp(false);
212 if (!initp)
213 {
214 std::sort(spectrum.begin(), spectrum.end(), svg::color_qf_lt_hue);
215 std::reverse(spectrum.begin(), spectrum.end());
216 initp = true;
217 }
218 return spectrum;
219}

222/// Random entry from array above.
223color_qi
224random_color(const uint startoffset = 0)
225{
226 auto& spectrum = active_spectrum();
227 const uint maxc = spectrum.size();
228 static std::mt19937_64 rg(std::random_device{}());
229 auto disti = std::uniform_int_distribution<>(startoffset, maxc - 1);
230 uint index = disti(rg);
231 return spectrum[index];
232}
233
234/// Loop through color array starting at position c.
235color_qi
236next_color(const color_qi klr)
237{
238 auto& spectrum = active_spectrum();
239 color_qi cnext = spectrum.front();
240 auto itr = std::find(spectrum.begin(), spectrum.end(), klr);
241 if (itr != spectrum.end())
242 {
243 ++itr;
244 if (itr == spectrum.end())
245 itr = spectrum.begin();
246 cnext = *itr;
247 }
248 return cnext;
249}
250
251/// Start at specified color bar entry point.
252color_qi
253start_at_color(const color_qi klr)
254{
255 static color_qi _S_klr = klr;
256 color_qi retk = _S_klr;
257 _S_klr = next_color(_S_klr);
258 return retk;
259}
