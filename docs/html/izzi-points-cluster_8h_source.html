32struct Point
33{
34 double x, y;
35
36 Point(double x = 0, double y = 0) : x(x), y(y) {}
37
38 double
39 distance(const Point& other) const
40 { return std::hypot(x - other.x, y - other.y); }
41
42 bool
43 operator==(const Point& other) const
44 { return x == other.x && y == other.y; }
45
46 Point
47 operator+(const Point& other) const
48 { return Point(x + other.x, y + other.y); }
49
50 Point
51 operator/(double scalar) const
52 { return Point(x / scalar, y / scalar); }
53};

79/// Reduce a set of points via a given cluster algorithm.

81{
82 double radius;

85public:
86 point_cluster(const vpoints& input_points, const double r)

89 /// Algorithm 1: Grid-based Clustering (Simple and Fast)
90 vwpoints
91 reduce_grid()
92 {
93 if (points.empty()) return {};
94
95 // Create grid cells
96 std::unordered_map<std::pair<int, int>, vpoints,
97 decltype([](const auto& p) {
98 return std::hash<int>{}(p.first) ^ (std::hash<int>{}(p.second) << 1);
99 })> grid;
100
101 for (const auto& point : points)
102 {
103 int grid_x = static_cast<int>(point.x / radius);
104 int grid_y = static_cast<int>(point.y / radius);
105 grid[{grid_x, grid_y}].push_back(point);
106 }
107
108 vwpoints result;
109 for (const auto& [cell, cell_points] : grid)
110 {
111 if (!cell_points.empty())
112 {
113 auto centroid = calculate_centroid(cell_points);
114 result.emplace_back(centroid.x, centroid.y, cell_points.size());
115 }
116 }
117
118 return result;
119 }

121 /// Algorithm 2: Hierarchical Agglomerative Clustering (More Accurate)
122 vwpoints
123 reduce_hierarchical_agglomerative()
124 {
125 if (points.empty()) return {};
126
127 std::vector<vpoints> clusters;
128 std::vector<bool> assigned(points.size(), false);
129
130 for (size_t i = 0; i < points.size(); ++i) {
131 if (assigned[i]) continue;
132
133 vpoints current_cluster;
134 std::queue<size_t> to_process;
135 to_process.push(i);
136 assigned[i] = true;
137
138 while (!to_process.empty()) {
139 size_t current_idx = to_process.front();
140 to_process.pop();
141 current_cluster.push_back(points[current_idx]);
142
143 // Find all unassigned points within cluster radius
144 for (size_t j = 0; j < points.size(); ++j) {
145 if (!assigned[j] && points[current_idx].distance(points[j]) <= radius) {
146 assigned[j] = true;
147 to_process.push(j);
148 }
149 }
150 }
151
152 if (!current_cluster.empty()) {
153 clusters.push_back(std::move(current_cluster));
154 }
155 }
156
157 return weigh_cluster(clusters);
158 }

160 /// Algorithm 3: K-Means Inspired with Distance Constraint
161 vwpoints
162 reduce_constrained_k_means(size_t max_iterations = 100)
163 {
164 if (points.empty()) return {};
165
166 auto centroids = initialize_centroids();
167 std::vector<vpoints> clusters(centroids.size());
168
169 for (size_t iter = 0; iter < max_iterations; ++iter) {
170 // Clear current clusters
171 for (auto& cluster : clusters) {
172 cluster.clear();
173 }
174
175 // Assign points to nearest centroid within cluster radius
176 for (const auto& point : points) {
177 double min_dist = std::numeric_limits<double>::max();
178 int best_cluster = -1;
179
180 for (size_t i = 0; i < centroids.size(); ++i) {
181 double dist = point.distance(centroids[i]);
182 if (dist <= radius && dist < min_dist) {
183 min_dist = dist;
184 best_cluster = static_cast<int>(i);
185 }
186 }
187
188 if (best_cluster != -1) {
189 clusters[best_cluster].push_back(point);
190 }
191 }
192
193 // Update centroids
194 vpoints new_centroids;
195 for (size_t i = 0; i < clusters.size(); ++i) {
196 if (!clusters[i].empty()) {
197 new_centroids.push_back(calculate_centroid(clusters[i]));
198 }
199 }
200
201 // Check for convergence
202 if (new_centroids.size() == centroids.size()) {
203 bool converged = true;
204 for (size_t i = 0; i < centroids.size(); ++i) {
205 if (centroids[i].distance(new_centroids[i]) > 1e-6) {
206 converged = false;
207 break;
208 }
209 }
210 if (converged) break;
211 }
212
213 centroids = std::move(new_centroids);
214 clusters.resize(centroids.size());
215 }
216
217 return weigh_cluster(clusters);
218 }

220 /// Algorithm 4: Voronoi-Based Clustering with Radius Constraint
221 vwpoints
222 reduce_voronoi(size_t max_iterations = 50)
223 {
224 if (points.empty()) return {};
225
226 // Step 1: Initialize sites using farthest point sampling
227 auto sites = initialize_centroids();
228 std::vector<voronoi_cell> cells;
229
230 for (const auto& site : sites) {
231 cells.emplace_back(site);
232 }
233
234 for (size_t iter = 0; iter < max_iterations; ++iter) {
235 // Clear current cell points
236 for (auto& cell : cells) {
237 cell.points.clear();
238 }
239
240 // Step 2: Assign each point to nearest site (Voronoi partitioning)
241 for (const auto& point : points) {
242 double min_dist = std::numeric_limits<double>::max();
243 size_t best_cell = 0;
244
245 for (size_t i = 0; i < cells.size(); ++i) {
246 double dist = point.distance(cells[i].site);
247 if (dist < min_dist) {
248 min_dist = dist;
249 best_cell = i;
250 }
251 }
252
253 // Only assign if within cluster radius of the site
254 if (min_dist <= radius) {
255 cells[best_cell].points.push_back(point);
256 }
257 }
258
259 // Step 3: Update sites to centroids of their Voronoi cells
260 bool converged = true;
261 for (size_t i = 0; i < cells.size(); ++i) {
262 if (!cells[i].points.empty()) {
263 Point new_site = calculate_centroid(cells[i].points);
264 if (cells[i].site.distance(new_site) > 1e-6) {
265 converged = false;
266 }
267 cells[i].site = new_site;
268 }
269 }
270
271 if (converged) break;
272
273 // Step 4: Remove empty cells and merge nearby sites
274 cells.erase(std::remove_if(cells.begin(), cells.end(),
275 [](const voronoi_cell& cell) { return cell.points.empty(); }),
276 cells.end());
277
278 // Merge sites that are too close
279 merge_close_sites(cells);
280 }
281
282 // Create final clusters by ensuring radius constraint
283 return constrain_voronoi_clusters(cells);
284 }

286private:
287 vpoints
288 initialize_centroids()
289 {
290 vpoints centroids;
291 if (!points.empty())
292 {
293 std::vector<bool> selected(points.size(), false);
294
295 // Start with a random point
296 std::random_device rd;
297 std::mt19937 gen(rd());
298 std::uniform_int_distribution<size_t> dist(0, points.size() - 1);
299
300 size_t first_idx = dist(gen);
301 centroids.push_back(points[first_idx]);
302 selected[first_idx] = true;
303
304 // Add centroids that are at least radius away from existing ones
305 for (size_t i = 0; i < points.size(); ++i)
306 {
307 if (selected[i])
308 continue;
309
310 bool far_enough = true;
311 for (const auto& centroid : centroids)
312 {
313 if (points[i].distance(centroid) < radius)
314 {
315 far_enough = false;
316 break;
317 }
318 }
319
320 if (far_enough)
321 {
322 centroids.push_back(points[i]);
323 selected[i] = true;
324 }
325 }
326 }
327 return centroids;
328 }

330 Point
331 calculate_centroid(const vpoints& cluster)
332 {
333 Point cp(0,0);
334 if (!cluster.empty())
335 {
336 double sum_x = 0, sum_y = 0;
337 for (const auto& point : cluster)
338 {
339 sum_x += point.x;
340 sum_y += point.y;
341 }
342 cp = Point(sum_x / cluster.size(), sum_y / cluster.size());
343 }
344 return cp;
345 }

347 vwpoints
348 weigh_cluster(const std::vector<vpoints>& clusters)
349 {
350 vwpoints result;
351 for (const auto& cluster : clusters) {
352 if (!cluster.empty()) {
353 auto centroid = calculate_centroid(cluster);
354 result.emplace_back(centroid.x, centroid.y, cluster.size());
355 }
356 }
357 return result;
358 }

360 void
361 merge_close_sites(std::vector<voronoi_cell>& cells)
362 {
363 for (size_t i = 0; i < cells.size(); ++i)
364 {
365 for (size_t j = i + 1; j < cells.size(); )
366 {
367 if (cells[i].site.distance(cells[j].site) < radius * 0.5)
368 {
369 // Merge cell j into cell i
370 auto& cpoints = cells[i].points;
371 cpoints.insert(cpoints.end(),
372 cells[j].points.begin(), cells[j].points.end());
373 cells[i].site = calculate_centroid(cpoints);
374
375 // Remove cell j
376 cells.erase(cells.begin() + j);
377 }
378 else
379 ++j;
380 }
381 }
382 }

384 vwpoints
385 constrain_voronoi_clusters(const std::vector<voronoi_cell>& cells)
386 {
387 vwpoints result;
388 for (const auto& cell : cells)
389 {
390 if (!cell.points.empty())
391 {
392 // Further split cells that violate radius constraint
393 auto subclusters = split_large_cell(cell);
394 for (const auto& subcluster : subclusters)
395 {
396 if (!subcluster.empty())
397 {
398 auto centroid = calculate_centroid(subcluster);
399 result.emplace_back(centroid, subcluster.size());
400 }
401 }
402 }
403 }
404
405 return result;
406 }

408 std::vector<vpoints>
409 split_large_cell(const voronoi_cell& cell)
410 {
411 std::vector<vpoints> subclusters;
412
413 if (cell.points.empty()) return subclusters;
414
415 // Check if cell needs splitting (points too far from centroid)
416 std::vector<bool> assigned(cell.points.size(), false);
417
418 for (size_t i = 0; i < cell.points.size(); ++i) {
419 if (assigned[i]) continue;
420
421 vpoints current_subcluster;
422 std::queue<size_t> to_process;
423 to_process.push(i);
424 assigned[i] = true;
425
426 while (!to_process.empty()) {
427 size_t current_idx = to_process.front();
428 to_process.pop();
429 current_subcluster.push_back(cell.points[current_idx]);
430
431 // Find all unassigned points within cluster radius
432 for (size_t j = 0; j < cell.points.size(); ++j) {
433 if (!assigned[j] &&
434 cell.points[current_idx].distance(cell.points[j]) <= radius) {
435 assigned[j] = true;
436 to_process.push(j);
437 }
438 }
439 }
440
441 if (!current_subcluster.empty()) {
442 subclusters.push_back(std::move(current_subcluster));
443 }
444 }
445
446 return subclusters;
447 }

450// Main function interface
451vwpoints
452clusterPoints(const vpoints& points, double radius,
453 const std::string& method)
454{
455 point_cluster clusterer(points, radius);
456
457 if (method == "grid") {
458 return clusterer.reduce_grid();
459 } else if (method == "hierarchical") {
460 return clusterer.reduce_hierarchical_agglomerative();
461 } else if (method == "kmeans") {
462 return clusterer.reduce_constrained_k_means();
463 } else if (method == "voronoi") {
464 return clusterer.reduce_voronoi();
465 } else {
466 throw std::invalid_argument("Unknown clustering method");
467 }
468}
