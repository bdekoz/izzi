39// Utility function, like regex_replace.
40inline void
41string_replace(std::string& target, const std::string& match,
42 const std::string& replace)
43{
44 size_t pos = 0;
45 while((pos = target.find(match, pos)) != std::string::npos)
46 {
47 target.replace(pos, match.length(), replace);
48 pos += replace.length();
49 }
50}

82/// Scale value from min to max on range (nfloor, nceil).
83double
84scale_value_on_range(const ssize_type value, const ssize_type min,
85 const ssize_type max,
86 const ssize_type nfloor, const ssize_type nceil)
87{
88 double rmultp(nceil - nfloor);
89 double valnum(value - min);
90 double valdenom(max - min);
91 double weightn = (rmultp * (valnum / valdenom)) + nfloor;
92 return weightn;
93}

96/// Find cartesian distance between two 2D points.
97space_type
98distance_cartesian(const point_2t& p1, const point_2t& p2)
99{
100 auto [ x1, y1 ] = p1;
101 auto [ x2, y2 ] = p2;
102 auto distancex = (x2 - x1) * (x2 - x1);
103 auto distancey = (y2 - y1) * (y2 - y1);
104 space_type distance = sqrt(distancex + distancey);
105 return distance;
106}

109// Does point p1 of radius r1 intersect point p2 with radius r2?
110// https://developer.mozilla.org x 2D_collision_detection
111bool
112detect_collision(const point_2t& p1, const int r1, const point_2t& p2,
113 const int r2)
114{
115 bool ret(false);
116 if (distance_cartesian(p1, p2) < r1 + r2)
117 ret = true;
118 return ret;
119}

131/// Conversion between point size to pixels given dpi density.
132double
133pt_to_px(const uint i = 1)
134{
135 // 1pt is equal to exactly 1/72th of an inch.
136 // On a 72dpi output device (display), this is 1 (aka 72 * 1/72).
137 // On a 90dpi output device (display), this is 1.25 (aka 90 * 1/72).
138 // On a 96dpi output device (display), this is 1.33 (aka 96 * 1/72).
139 // constexpr double dpimult = (get_dpi() / 72);
140
141 // Or 18 pt -> 13.5 px -> 1.33
142 // Or 30 pt -> 22.5 px -> 1.33
143 constexpr double dpimult = 1.33;
144 return std::lround(i * dpimult);
145}
