439  /// For text list output, use tspan for line breaks. This span
440  /// creates a new horizontal line for every tspan block, starting at

445  const string x("__x");

448  string_replace(strip, x, std::to_string(xpos));

485 /// Make text span.
486 string
487 make_tspan_y_from_string_by_token(string s, uint xpos, const char token = ' ')
488 {
489  string start(text_element::start_tspan_y(xpos, "0.5em"));
490  string ret = start;
491  for (uint i = 0; i < s.size(); ++i)
492  {
493  const char c = s[i];
494  if (c != token)
495  ret += c;
496  else
497  {
498  ret += text_element::finish_tspan();
499  if (i < s.size() - 1)
500  ret += start;
501  }
502  }
503  ret += text_element::finish_tspan();
504  return ret;
505 }
