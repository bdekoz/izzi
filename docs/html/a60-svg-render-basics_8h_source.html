22double
23scale_proportional_to_area(double radius, double weight)
24{
25 // Scale proportional to area of generated circle.
26 // Ï€ = 3.14159265358979323846
27 const double pi(22/7);
28 double a1 = radius * radius * pi;
29 double ap = a1 * weight;
30 double rpa = std::sqrt(ap/pi);
31 return rpa;
32}

142/// Text of maxlen length, overflow goes on line below.
143uint
144text_line_n(svg_element& obj, const point_2t origin, const string text,
145 svg::typography typo, const int sz, const uint maxlen)
146{
147 auto [ x, y ] = origin;
148
149 string textcut(text);
150 while (textcut.size() > maxlen)
151 {
152 // Find last space character in the specified maxium range, aka mark.
153 auto sppos = textcut.find_last_of(k::space, maxlen);
154 if (sppos == string::npos)
155 sppos = maxlen;
156 else
157 {
158 // Cut after space (mark).
159 sppos += 1;
160 }
161
162 string namesubs = textcut.substr(0, sppos);
163 sized_text(obj, typo, sz, namesubs, x, y);
164 textcut = textcut.substr(sppos);
165 y += sz;
166 }
167 sized_text(obj, typo, sz, textcut, x, y);
168 return y;
169}

172/// Text of maxlen length rotated, overflow goes on line below.
173uint
174text_line_n_r(svg_element& obj, const point_2t origin, const string text,
175 svg::typography typo, const uint sz, const uint maxlen,
176 const uint lettingsz = 0)
177{
178 const auto [ x, y ] = origin;
179 const double line_max = std::ceil(double(text.size()) / maxlen);
180 const uint lines(line_max);
181
182 uint xmin(x);
183 string textcut(text);
184
185 uint linen(0);
186 while (linen < lines)
187 {
188 // Find x offset, with first line being the max length above the
189 // origin, working down. The last line is at the origin.
190 auto xp = x - ((sz + lettingsz) * (lines - linen - 1));
191
192 size_t epos(0);
193 if (textcut.size() < maxlen)
194 epos = textcut.size();
195 else
196 {
197 // Find last space character in the specified maxium range, aka mark.
198 auto sppos = textcut.find_last_of(k::space, maxlen);
199 if (sppos == string::npos)
200 sppos = maxlen;
201 else
202 {
203 // Cut after space (mark).
204 sppos += 1;
205 }
206 epos = sppos;
207 }
208
209 string namesubs = textcut.substr(0, epos);
210 sized_text_r(obj, typo, sz, namesubs, xp, y, -90);
211 textcut = textcut.substr(epos);
212 ++linen;
213 xmin = std::min(xmin, uint(xp));
214 }
215 return xmin;
216}

328/// Draws a ring centered at origin of radius r, with outer and inner
329/// radial gradient of blurspace in each direction.
330/// klr == fade from
331/// fadeklr == fade to. Background is transparent if none.
332void
333point_to_ring_halo(svg_element& obj, const point_2t origin,
334 const space_type radius, const double blurspace,
335 const svg::color klr,
336 const svg::color fadeklr = color::none,
337 const double opacity = 1)
338{
339 using atype = decltype(obj._M_area)::atype;
340
341 auto [ xd, yd ] = origin;
342 const atype x(xd);
343 const atype y(yd);
344
345 // outer ring == upper bound, radius + variance.
346 const double oring = radius + blurspace;
347
348 // inner ring == lower bound, radius - variance.
349 const double iring = radius - blurspace;
350
351 // mangled args for name.
352 std::ostringstream oss;
353 oss << "x" << std::to_string(x) << k::hyphen
354 << "y" << std::to_string(y) << k::hyphen
355 << "r" << std::to_string(radius) << k::hyphen
356 << "blurspace" << std::to_string(blurspace);
357 const string mangle(oss.str());
358
359 // outer
360 // strategy: make a bigger circle cprime, then do a radial gradient to it
361 // starting gradient from color at radius to 100% white/tranparent at cprime.
362 const string rgrado_name(string("radialout") + k::hyphen + mangle);
363 radial_gradient rgrado;
364 rgrado.start_element(rgrado_name);
365 rgrado.stop(rgrado.offset_percentage(radius, oring), fadeklr, 0);
366 rgrado.stop(rgrado.offset_percentage(radius, oring), klr, opacity);
367 rgrado.stop("100%", color::white, 0);
368 rgrado.finish_element();
369 obj.add_element(rgrado);
370
371 circle_element co;
372 circle_element::data dco = { x, y, atype(oring) };
373 co.start_element();
374 co.add_data(dco);
375 co.add_fill(rgrado_name);
376 co.finish_element();
377 obj.add_element(co);
378
379 // inner
380 // strategy: make a smaller circle cprime, then do a radial gradient from it
381 // starting gradient from white/transparent at cprime to color at r.
382 const string rgradi_name(string("radialin") + k::hyphen + mangle);
383 radial_gradient rgradi;
384 rgradi.start_element(rgradi_name);
385 rgradi.stop(rgradi.offset_percentage(iring, radius), fadeklr, 0);
386 rgradi.stop("100%", klr, opacity);
387 rgradi.finish_element();
388 obj.add_element(rgradi);
389
390 // Float/Int conversion and rounding, add one to radius to close gap.
391 circle_element ci;
392 circle_element::data dci = { x, y, radius + 1 };
393 ci.start_element();
394 ci.add_data(dci);
395 ci.add_fill(rgradi_name);
396 ci.finish_element();
397 obj.add_element(ci);
398}

401/// Lines radiating from center point (x,y).
402void
403point_2d_to_ray(svg_element& obj, double x, double y, svg::style s,
404 space_type r = 4, const uint nrays = 10)
405{
406 using atype = decltype(obj._M_area)::atype;
407
408 // End points on the ray.
409 // Pick a random ray, use an angle in the range [0, 2pi].
410 static std::mt19937_64 rg(std::random_device{}());
411 auto distr = std::uniform_real_distribution<>(0.0, 2 * 22/7);
412 auto disti = std::uniform_int_distribution<>(-3, 3);
413
414 for (uint i = 0; i < nrays; ++i)
415 {
416 double theta = distr(rg);
417 double rvary = disti(rg);
418
419 atype xe = x + (r + rvary) * std::cos(theta);
420 atype ye = y + (r + rvary) * std::sin(theta);
421
422 line_element::data dr = { atype(x), xe, atype(y), ye };
423 line_element ray;
424 ray.start_element();
425 ray.add_data(dr);
426 ray.add_style(s);
427 ray.finish_element();
428 obj.add_element(ray);
429 }
430}

457/// Angle in radians.
458point_2t
459get_circumference_point_r(const double angler, const double r,
460 const point_2t origin)
461{
462 auto [ cx, cy ] = origin;
463 double x(cx + (r * std::cos(angler)));
464 double y(cy - (r * std::sin(angler)));
465 return std::make_tuple(x, y);
466}

470point_2t
471get_circumference_point_d(const double ad, const double r,
472 const point_2t origin)
473{
474 double angler = (k::pi / 180.0) * ad;
475 return get_circumference_point_r(angler, r, origin);

479/// Zero degrees is top, going clockwise (cw).
480double
481zero_angle_north_cw(double angled)
482{
483 // Change rotation to CW instead of CCW (or anti-clockwise).
484 angled = 360 - angled;
485
486 // Rotate 90 CCW, so that the first element will be at the top
487 // vertical axis, instead of the right middle axis.
488 angled += 90;
489
490 return angled;
491}

506/// Make single path segment.
507string
508make_path_data_from_points(const vrange& lpoints)
509{
510 std::ostringstream ossa;
511 for (uint i = 0; i < lpoints.size(); ++i)
512 {
513 auto [x, y ] = lpoints[i];
514 // SVG path_element.
515 // start at "M x y" and
516 // each subsequent line segment is of form "L x y"
517 if (i == 0)
518 ossa << "M" << k::space;
519 else
520 ossa << "L" << k::space;
521 ossa << x << svg::k::space << y << k::space;
522 }
523 return ossa.str();
524}

527/// Draw path given serialized path data.
528/// Assumes pinstripe, ie top and bottom line layers.
529/// top style defaults: fill opac(0), stroke opac(1), stroke sz 1
530/// bottom style defaults: fill opac(0), stroke opac(1), stroke sz 1.25
531string
532draw_path_data(const string& pathda,
533 const style& topstyl, const style& bottomstyl)
534{
535 std::ostringstream oss;
536
537 // Draw path with this endpoint.
538 path_element::data da = { pathda, 0 };
539
540 // Draw colored line and outline. Draw outline first, so
541 // it will be on the lower layer in the SVG file.
542 if (to_string(bottomstyl._M_stroke_color) != to_string(svg::color::none))
543 {
544 path_element pebase;
545 pebase.start_element();
546 pebase.add_data(da);
547 pebase.add_style(bottomstyl);
548 pebase.finish_element();
549 oss << pebase.str() << k::space;
550 }
551
552 path_element pe;
553 pe.start_element();
554 pe.add_data(da);
555 pe.add_style(topstyl);
556 pe.finish_element();
557 oss << pe.str() << k::space;
558
559 return oss.str();
560}

563/// Center a triangle at this point.
564path_element
565make_path_triangle(const point_2t origin, svg::style s,
566 const double r = 4, const double angle = 120)
567{
568 // Find points: orig, orig + (120 x 1), orig + (120 x 2).
569 double zo = zero_angle_north_cw(angle);
570 point_2t p1 = get_circumference_point_d(zo, r, origin);
571 point_2t p2 = get_circumference_point_d(zo + (angle * 1), r, origin);
572 point_2t p3 = get_circumference_point_d(zo + (angle * 2), r, origin);
573 vrange pointz = { p1, p2, p3, p1 };
574 string pathda = make_path_data_from_points(pointz);
575
576 // Make closed path.
577 path_element tri(true);
578 path_element::data pthdata = { pathda, 0 };
579 string argname = std::to_string(r) + k::hyphen + std::to_string(angle);
580 tri.start_element("triangle-" + argname);
581 tri.add_data(pthdata);
582 tri.add_style(s);
583 tri.finish_element();
584 return tri;
585}

598/// Center an octogon at this point.
599/// radius 4 is pixels to draw out from center point.
600/// pointsn is number of points to draw (8 for octogon)
601path_element
602make_path_octogon(const point_2t origin, svg::style s, const double r = 4,
603 const uint pointsn = 8)
604{
605 // Find points: orig, orig + (120 x 1), orig + (120 x 2).
606 const double angle(360.0/8);
607 double zo = zero_angle_north_cw(angle);
608
609 // Eight points on a circle, connnected.
610 vrange pointz;
611 for (uint i = 0; i < pointsn; ++i)
612 {
613 point_2t p = get_circumference_point_d(zo + (angle * i), r, origin);
614 pointz.push_back(p);
615 }
616
617 // Final point to close path.
618 pointz.push_back(pointz.front());
619 string pathda = make_path_data_from_points(pointz);
620
621 // Make closed path.
622 path_element oct(true);
623 path_element::data pthdata = { pathda, 0 };
624 oct.start_element("octogon-" + std::to_string(r));
625 oct.add_data(pthdata);
626 oct.add_style(s);
627 oct.finish_element();
628 return oct;
629}

661/// Make closed path between two points and the center of a circle of radius r.
662/// Points like: get_circumference_point_d(zero_angle_north_cw(0), r, origin)
663string
664make_path_arc_closed(const point_2t& origin, const point_2t& start,
665 const point_2t& end, const space_type r,
666 const int arcflag = 0, const int sweepflag = 0)
667{
668 // Define path as starting at origin, line to circumference point start,
669 // arc to circumfernce point end, line back to origin.
670 // A rx ry x-axis-rotation large-arc-flag sweep-flag x y
671 // where (large) arc flag is true if arc angle delta is > 180
672 // where sweep flag is
673 // true if outer (movement CW)
674 // false if inner (CCW).
675 std::ostringstream oss;
676 oss << "M" << k::space << to_string(origin) << k::space;
677 oss << "L" << k::space << to_string(start) << k::space;
678 oss << "A" << k::space;
679 oss << std::to_string(r) << k::space << std::to_string(r) << k::space;
680 oss << 0 << k::space << arcflag << k::space << sweepflag << k::space;
681 oss << to_string(end) << k::space;
682 oss << "L" << k::space << to_string(origin) << k::space;
683 return oss.str();
684}

705 // move around as if making a plus sign.
706 const auto [ xo, yo ] = origin;
707
708 // Move to top left of origin, start here.
709 const double whalf(width / 2);
710 const int lenw = len - whalf;
711 const auto x = xo - whalf;
712 const auto y = yo - whalf;
713
714 std::ostringstream oss;
715 oss << "M" << k::space << x << k::comma << y << k::space;
716
717 // left
718 oss << "H" << k::space << x - lenw << k::space;
719 oss << "V" << k::space << y + width << k::space;
720 oss << "H" << k::space << x << k::space;
721
722 // bottom
723 oss << "V" << k::space << y + lenw + width << k::space;
724 oss << "H" << k::space << x + width << k::space;
725 oss << "V" << k::space << y + width << k::space; // bottom part
726
727 // right
728 oss << "H" << k::space << x + lenw + width << k::space;
729 oss << "V" << k::space << y << k::space;
730 oss << "H" << k::space << x + width << k::space;
731
732 // top
733 oss << "V" << k::space << y - lenw << k::space;
734 oss << "H" << k::space << x << k::space;
735 oss << "V" << k::space << y << k::space;
736
737 return oss.str();
738}
