23double
24scale_proportional_to_area(double radius, double weight)
25{
26 // Scale proportional to area of generated circle.
27 // Ï€ = 3.14159265358979323846
28 const double pi(22/7);
29 double a1 = radius * radius * pi;
30 double ap = a1 * weight;
31 double rpa = std::sqrt(ap/pi);
32 return rpa;
33}

112void
113styled_text_link(svg_element& obj, const string text, const point_2t origin,
114 const typography typo, const string uri)
115{
116 using atype = decltype(obj._M_area)::atype;
117 auto [ x, y ] = origin;
118 text_element::data dt = { atype(x), atype(y), text, typo };
119 text_element t;
120 t.start_element();
121 t.add_data(dt);
122 t.finish_element();
123
124 // Convert uri to utf8 so that it can be inserted into svg_element
125 // Fucking '&', mostly.
126 // string uriconv = convert_to_utf8(uri);
127 string uriconv(uri);
128 size_t pos = 0;
129 while ((pos = uriconv.find('&', pos)) != std::string::npos)
130 {
131 uriconv.replace(pos, 1, "&amp;");
132 pos += 5;
133 }
134
135 string astart = "<a href=";
136 astart += k::quote;
137 astart += uriconv;
138 astart += k::quote;
139 astart +=">";
140 astart += k::newline;
141 obj.add_raw(astart);
142
143 obj.add_element(t);
144
145 string afinish = "</a>";
146 afinish += k::newline;
147 obj.add_raw(afinish);

183/// Text of maxlen length, overflow goes on line below.
184uint
185text_line_n(svg_element& obj, const point_2t origin, const string text,
186 const svg::typography typo, const int sz, const uint maxlen)
187{
188 auto [ x, y ] = origin;
189
190 string textcut(text);
191 while (textcut.size() > maxlen)
192 {
193 // Find last space character in the specified maxium range, aka mark.
194 auto sppos = textcut.find_last_of(k::space, maxlen);
195 if (sppos == string::npos)
196 sppos = maxlen;
197 else
198 {
199 // Cut after space (mark).
200 sppos += 1;
201 }
202
203 string namesubs = textcut.substr(0, sppos);
204 sized_text(obj, typo, sz, namesubs, x, y);
205 textcut = textcut.substr(sppos);
206 y += sz;
207 }
208 sized_text(obj, typo, sz, textcut, x, y);
209 return y;
210}

213/// Text of maxlen length rotated, overflow goes on line below.
214uint
215text_line_n_r(svg_element& obj, const point_2t origin, const string text,
216 const svg::typography typo, const uint sz, const uint maxlen,
217 const uint lettingsz = 0)
218{
219 const auto [ x, y ] = origin;
220 const double line_max = std::ceil(double(text.size()) / maxlen);
221 const uint lines(line_max);
222
223 uint xmin(x);
224 string textcut(text);
225
226 uint linen(0);
227 while (linen < lines)
228 {
229 // Find x offset, with first line being the max length above the
230 // origin, working down. The last line is at the origin.
231 auto xp = x - ((sz + lettingsz) * (lines - linen - 1));
232
233 size_t epos(0);
234 if (textcut.size() < maxlen)
235 epos = textcut.size();
236 else
237 {
238 // Find last space character in the specified maxium range, aka mark.
239 auto sppos = textcut.find_last_of(k::space, maxlen);
240 if (sppos == string::npos)
241 sppos = maxlen;
242 else
243 {
244 // Cut after space (mark).
245 sppos += 1;
246 }
247 epos = sppos;
248 }
249
250 string namesubs = textcut.substr(0, epos);
251 sized_text_r(obj, typo, sz, namesubs, xp, y, -90);
252 textcut = textcut.substr(epos);
253 ++linen;
254 xmin = std::min(xmin, uint(xp));
255 }
256 return xmin;
257}

389/// Draws a ring centered at origin of radius r, with outer and inner
390/// radial gradient of blurspace in each direction.
391/// klr == fade from
392/// fadeklr == fade to. Background is transparent if none.
393void
394point_to_ring_halo(svg_element& obj, const point_2t origin,
395 const space_type radius, const double blurspace,
396 const svg::color klr,
397 const svg::color fadeklr = color::none,
398 const double opacity = 1)
399{
400 using atype = decltype(obj._M_area)::atype;
401
402 auto [ xd, yd ] = origin;
403 const atype x(xd);
404 const atype y(yd);
405
406 // outer ring == upper bound, radius + variance.
407 const double oring = radius + blurspace;
408
409 // inner ring == lower bound, radius - variance.
410 const double iring = radius - blurspace;
411
412 // mangled args for name.
413 std::ostringstream oss;
414 oss << "x" << std::to_string(x) << k::hyphen
415 << "y" << std::to_string(y) << k::hyphen
416 << "r" << std::to_string(radius) << k::hyphen
417 << "blurspace" << std::to_string(blurspace);
418 const string mangle(oss.str());
419
420 // outer
421 // strategy: make a bigger circle cprime, then do a radial gradient to it
422 // starting gradient from color at radius to 100% white/tranparent at cprime.
423 const string rgrado_name(string("radialout") + k::hyphen + mangle);
424 radial_gradient rgrado;
425 rgrado.start_element(rgrado_name);
426 rgrado.stop(rgrado.offset_percentage(radius, oring), fadeklr, 0);
427 rgrado.stop(rgrado.offset_percentage(radius, oring), klr, opacity);
428 rgrado.stop("100%", color::white, 0);
429 rgrado.finish_element();
430 obj.add_element(rgrado);
431
432 circle_element co;
433 circle_element::data dco = { x, y, atype(oring) };
434 co.start_element();
435 co.add_data(dco);
436 co.add_fill(rgrado_name);
437 co.finish_element();
438 obj.add_element(co);
439
440 // inner
441 // strategy: make a smaller circle cprime, then do a radial gradient from it
442 // starting gradient from white/transparent at cprime to color at r.
443 const string rgradi_name(string("radialin") + k::hyphen + mangle);
444 radial_gradient rgradi;
445 rgradi.start_element(rgradi_name);
446 rgradi.stop(rgradi.offset_percentage(iring, radius), fadeklr, 0);
447 rgradi.stop("100%", klr, opacity);
448 rgradi.finish_element();
449 obj.add_element(rgradi);
450
451 // Float/Int conversion and rounding, add one to radius to close gap.
452 circle_element ci;
453 circle_element::data dci = { x, y, radius + 1 };
454 ci.start_element();
455 ci.add_data(dci);
456 ci.add_fill(rgradi_name);
457 ci.finish_element();
458 obj.add_element(ci);
459}

462/// Lines radiating from center point (x,y).
463void
464point_2d_to_ray(svg_element& obj, double x, double y, svg::style s,
465 space_type r = 4, const uint nrays = 10)
466{
467 using atype = decltype(obj._M_area)::atype;
468
469 // End points on the ray.
470 // Pick a random ray, use an angle in the range [0, 2pi].
471 static std::mt19937_64 rg(std::random_device{}());
472 auto distr = std::uniform_real_distribution<>(0.0, 2 * 22/7);
473 auto disti = std::uniform_int_distribution<>(-3, 3);
474
475 for (uint i = 0; i < nrays; ++i)
476 {
477 double theta = distr(rg);
478 double rvary = disti(rg);
479
480 atype xe = x + (r + rvary) * std::cos(theta);
481 atype ye = y + (r + rvary) * std::sin(theta);
482
483 line_element::data dr = { atype(x), xe, atype(y), ye };
484 line_element ray;
485 ray.start_element();
486 ray.add_data(dr);
487 ray.add_style(s);
488 ray.finish_element();
489 obj.add_element(ray);
490 }
491}

518/// Angle in radians.
519point_2t
520get_circumference_point_r(const double angler, const double r,
521 const point_2t origin)
522{
523 auto [ cx, cy ] = origin;
524 double x(cx + (r * std::cos(angler)));
525 double y(cy - (r * std::sin(angler)));
526 return std::make_tuple(x, y);
527}

531point_2t
532get_circumference_point_d(const double ad, const double r,
533 const point_2t origin)
534{
535 double angler = (k::pi / 180.0) * ad;
536 return get_circumference_point_r(angler, r, origin);

540/// Zero degrees is top, going clockwise (cw).
541double
542zero_angle_north_cw(double angled)
543{
544 // Change rotation to CW instead of CCW (or anti-clockwise).
545 angled = 360 - angled;

567/// Make single path segment.
568string
569make_path_data_from_points(const vrange& lpoints)
570{
571 std::ostringstream ossa;
572 for (uint i = 0; i < lpoints.size(); ++i)
573 {
574 auto [x, y ] = lpoints[i];
575 // SVG path_element.
576 // start at "M x y" and
577 // each subsequent line segment is of form "L x y"
578 if (i == 0)
579 ossa << "M" << k::space;
580 else
581 ossa << "L" << k::space;
582 ossa << x << svg::k::space << y << k::space;
583 }
584 return ossa.str();
585}

612/// Center a triangle at this point.
613path_element
614make_path_triangle(const point_2t origin, const svg::style styl,
615 const double r = 4, const double angle = 120)
616{
617 // Find points: orig, orig + (120 x 1), orig + (120 x 2).
618 double zo = zero_angle_north_cw(angle);
619 point_2t p1 = get_circumference_point_d(zo, r, origin);
620 point_2t p2 = get_circumference_point_d(zo + (angle * 1), r, origin);
621 point_2t p3 = get_circumference_point_d(zo + (angle * 2), r, origin);
622 vrange pointz = { p1, p2, p3, p1 };
623 string pathda = make_path_data_from_points(pointz);
624
625 string argname = std::to_string(r) + k::hyphen + std::to_string(angle);
626 const string id = "triangle-" + argname;
627
628 path_element::data pthdata = { pathda, 0 };
629 path_element tri = make_path(pathda, styl, id);
630 return tri;
631}

644/// Center an octogon at this point.
645/// radius 4 is pixels to draw out from center point.
646/// pointsn is number of points to draw (8 for octogon)
647path_element
648make_path_octogon(const point_2t origin, const svg::style styl,
649 const double r = 4, const uint pointsn = 8)
650{
651 // Find points: orig, orig + (120 x 1), orig + (120 x 2).
652 const double angle(360.0/8);
653 double zo = zero_angle_north_cw(angle);
654
655 // Eight points on a circle, connnected.
656 vrange pointz;
657 for (uint i = 0; i < pointsn; ++i)
658 {
659 point_2t p = get_circumference_point_d(zo + (angle * i), r, origin);
660 pointz.push_back(p);
661 }
662
663 // Final point to close path.
664 pointz.push_back(pointz.front());
665 string pathda = make_path_data_from_points(pointz);
666
667 const string id = "octogon-" + std::to_string(r);
668 path_element oct = make_path(pathda, styl, id);
669 return oct;
670}

683/// Make path segment between two points on a circumference of radius r.

686make_path_arc_circumference(const point_2t& start, const point_2t& end,

692 std::ostringstream oss;

697 oss << to_string(end) << k::space;

702/// Make closed path between two points and the center of a circle of radius r.
703/// Points like: get_circumference_point_d(zero_angle_north_cw(0), r, origin)
704string
705make_path_arc_closed(const point_2t& origin, const point_2t& start,
706 const point_2t& end, const space_type r,
707 const int arcflag = 0, const int sweepflag = 0)
708{
709 // Define path as starting at origin, line to circumference point start,
710 // arc to circumfernce point end, line back to origin.
711 // A rx ry x-axis-rotation large-arc-flag sweep-flag x y
712 // where (large) arc flag is true if arc angle delta is > 180
713 // where sweep flag is
714 // true if outer (movement CW)
715 // false if inner (CCW).
716 std::ostringstream oss;
717 oss << "M" << k::space << to_string(origin) << k::space;
718 oss << "L" << k::space << to_string(start) << k::space;
719 oss << "A" << k::space;
720 oss << std::to_string(r) << k::space << std::to_string(r) << k::space;
721 oss << 0 << k::space << arcflag << k::space << sweepflag << k::space;
722 oss << to_string(end) << k::space;

741/// Plus or x tilt mark as closed path that can be filled.
742path_element
743make_path_center_mark(const point_2t& origin, const svg::style styl,
744 const int len, const int width)
745{
746 // Define path as starting at origin, move half width up and to left then
747 // move around as if making a plus sign.
748 const auto [ xo, yo ] = origin;
749
750 // Move to top left of origin, start here.
751 const double whalf(width / 2);
752 const int lenw = len - whalf;
753 const auto x = xo - whalf;
754 const auto y = yo - whalf;
755
756 std::ostringstream oss;
757 oss << "M" << k::space << x << k::comma << y << k::space;
758
759 // left
760 oss << "H" << k::space << x - lenw << k::space;
761 oss << "V" << k::space << y + width << k::space;
762 oss << "H" << k::space << x << k::space;
763
764 // bottom
765 oss << "V" << k::space << y + lenw + width << k::space;
766 oss << "H" << k::space << x + width << k::space;
767 oss << "V" << k::space << y + width << k::space; // bottom part
768
769 // right
770 oss << "H" << k::space << x + lenw + width << k::space;
771 oss << "V" << k::space << y << k::space;
772 oss << "H" << k::space << x + width << k::space;
773
774 // top
775 oss << "V" << k::space << y - lenw << k::space;
776 oss << "H" << k::space << x << k::space;
777 oss << "V" << k::space << y << k::space;
778
779 const string pathdata = oss.str();
780
781 string id("center-mark-");
782 string attr(std::to_string(width) + "-" + std::to_string(len));
783 id += attr;
784
785 path_element cm = make_path(pathdata, styl, id);
786 return cm;
787}
