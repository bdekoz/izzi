24  /**
25  Selected or Active in render area.
26  Make discrete element or layer (visible, outline, etc) if true.
27  Used as a (visibility, outline, etc.) bitmask
28  */
29  enum class select : uint
30  {
31  none = 1u << 0, ///< nothing
32  cartography = 1u << 1, ///< cartographic elements
33  vector = 1u << 2, ///< svg path, circle, rectangle, etc.
34  cloud = 1u << 3, ///< opacity color fill version of vector
35  blur = 1u << 4, ///< blur or gradient version of vector
36  echo = 1u << 5, ///< b & w outline version of vector
37  text = 1u << 6, ///< title, metadata, header
38  legend = 1u << 7, ///< text and symbol legends
39  exitnode = 1u << 8, ///< tor exit node infrastructure
40  telecom = 1u << 9, ///< telecom infrastructure
41  glyph = 1u << 10, ///< glyph
42  image = 1u << 11, ///< image
43  svg = 1u << 12, ///< svg element, perhaps nested
44  alt = 1u << 13, ///< alternate use specified in situ
45  background = 1u << 14, ///< background elements
46  raster = 1u << 15, ///< raster (pixel) elements
47  top = 1u << 16, ///< top
48  bottom = 1u << 17, ///< bottom
49  inner = 1u << 18, ///< inner
50  outer = 1u << 19, ///< outer

268 /// Add value to cache with base style of styl, colors klr, visibility vis.
269 void
270 add_to_id_render_state_cache(const string id, const style styl,
271  const k::select vis)
272 {
273  id_render_state_umap& cache = get_id_render_state_cache();

281 /// Given identifier/name/id, get corresponding id_render_state from cache.
282 const id_render_state
283 get_id_render_state(const string id)
284 {
285  const id_render_state_umap& cache = get_id_render_state_cache();
286 
287  id_render_state ret;
288  if (cache.count(id) == 1)
289  {
290  auto iter = cache.find(id);
291  ret = iter->second;
292  }
293  else
294  {
295  // If default string is set in the cache, use it.
296  if (cache.count("") == 1)
297  {
298  auto iter = cache.find("");
299  ret = iter->second;
300  }
301  }
302  return ret;
303 }

306 /// Roll through render states given in values squentially,
307 /// index starts with zero.
308 const id_render_state
309 traverse_states(const strings& values)
310 {
311  static uint indx(0);
312  string value;
313  if (indx < values.size())
314  {
315  value = values[indx];
316  ++indx;
317  }
318  else
319  {
320  // Start over.
321  indx = 0;
322  value = values[indx];
323  }
324  return get_id_render_state(value);
325 }
