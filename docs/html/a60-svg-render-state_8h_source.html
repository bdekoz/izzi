22 /**
23 Selected or Active in render area.
24 Make discrete element or layer (visible, outline, etc) if true.
25 Used as a (visibility, outline, etc.) bitmask

27 enum class select
28 {
29 none = 1u << 0, ///< nothing
30 cartography = 1u << 1, ///< cartographic elements
31 vector = 1u << 2, ///< svg path, circle, rectangle, etc.
32 blur = 1u << 3, ///< blur or gradient version of vector
33 echo = 1u << 4, ///< b & w outline version of vector
34 text = 1u << 5, ///< metadata, header
35 title = 1u << 6, ///< title

44 glyph = 1u << 15, ///< glyph

46 svg = 1u << 17, ///< svg element, perhaps nested

49 raster = 1u << 20, ///< raster (pixel) elements

117 /**
118 Some high-level nobs for rendering: scale tuning.
119
120 small == defense distributed == 70k-300k
121 medium == westworld == 2-4M
122 large == stranger things == 10M+
123
124 [1-5]xsmall == reduction from least to greatest
125 1x == baseline
126 [1-5]xlarge == enlargement from least to greatest
127 */
128 enum class scale
129 {
130 r5s, r4s, r3s, r2s, r1s,
131 xxsmall, xsmall,
132 small,
133 medium, baseline,
134 large,
135 xlarge, xxlarge,
136 e1s, e2s , e3s, e4s, e5s,
137 };

151 enum_map[scale::r3s] = "reduce-3s";

153 enum_map[scale::r1s] = "reduce-1s";
154 enum_map[scale::e5s] = "enlarge-5s";
155 enum_map[scale::e4s] = "enlarge-4s";
156 enum_map[scale::e3s] = "enlarge-3s";
157 enum_map[scale::e2s] = "enlarge-2s";
158 enum_map[scale::e1s] = "enlarge-1s";
159 enum_map[scale::medium] = "medium";
160 enum_map[scale::baseline] = "baseline";

162 enum_map[scale::xsmall] = "xsmall";

167 }

172/// Settings for glyph, graphic, chart, and collection object render.
173struct render_state_base
174{
175 select visible_mode;
176
177 bool
178 is_visible(const select v) const
179 {
180 using __utype = typename std::underlying_type<select>::type;
181 __utype isv(static_cast<__utype>(visible_mode & v));
182 return isv > 0;
183 }
184
185 render_state_base(const select m = select::none) : visible_mode(m)
186 { }
187};

190/// Collection derived states.
191struct collection_rstate : public render_state_base
192{
193 select outline_mode;
194 scale scale_mode;
195
196 double opacity;
197 int text_scale_max;
198
199 // Values with similar geo coordinates counted in as one meta coordinate.
200 bool weigh;
201
202 // Genderate colors from color bands (true) or use finite colors (false).
203 bool color_generated;
204
205 // Alternate output naming mode via scheme (true = info, false = map)
206 bool alt;
207
208 collection_rstate(const double o = 0.10,
209 const scale rscale = scale::medium)
210 : render_state_base(select::none), outline_mode(select::none),
211 scale_mode(rscale), opacity(o), text_scale_max(0),
212 weigh(false), color_generated(true), alt(false)
213 { }
214
215 collection_rstate(const collection_rstate&) = default;
216
217 string
218 mangle() const
219 {
220 string mangled;
221
222 bool foundp = false;
223
224 const uint vi = static_cast<uint>(visible_mode);
225 if (vi > 0)
226 {
227 mangled += "visible-mode-";
228 mangled += std::to_string(vi);
229 foundp = true;
230 }
231
232 const uint oi = static_cast<uint>(outline_mode);
233 if (oi > 0)
234 {
235 if (foundp)
236 mangled += k::hyphen;
237 else
238 foundp = true;
239 mangled += "outline-mode-";
240 mangled += std::to_string(oi);
241 }
242
243 if (opacity > 0)
244 {
245 if (foundp)
246 mangled += k::hyphen;
247 else
248 foundp = true;
249 mangled += "opacity-";
250 const double o(opacity * 100);
251 const uint oi = static_cast<uint>(o);
252 mangled += std::to_string(oi);
253 }
254
255 return mangled;
256 }
257};

290/// Named render state.
291/// Datum to take id string and tie it to visual representation.
292struct id_rstate: public render_state_base
293{
294 style styl; // Overrides render_state_base.opacity
295 string name;
296 double rotate; // Degrees to rotate to origin (CW), if any.
297 double multiple; // Scale factor, if any.
298
299 static style dstyl;
300
301 explicit
302 id_rstate(const style s = dstyl, const string f = "",
303 const double angle = 0.0, const double scale = 1.0)
304 : styl(s), name(f), rotate(angle), multiple(scale) { }
305
306 id_rstate(const id_rstate&) = default;
307 id_rstate& operator=(const id_rstate&) = default;
308};

323/// Add value to cache with base style of styl, colors klr, visibility vis.
324void
325add_to_id_rstate_cache(const string id, const style styl,
326 const select vis)
327{
328 id_rstate_umap& cache = get_id_rstate_cache();
329
330 id_rstate state(styl, id);
331 set_select(state.visible_mode, vis);
332 cache.insert(std::make_pair(id, state));
333}
334
335
336/// Given identifier/name/id, get corresponding id_rstate from cache.
337const id_rstate
338get_id_rstate(const string id)
339{
340 const id_rstate_umap& cache = get_id_rstate_cache();
341
342 id_rstate ret;
343 if (cache.count(id) == 1)
344 {
345 auto iter = cache.find(id);
346 ret = iter->second;
347 }
348 else
349 {
350 // If default string is set in the cache, use it.
351 if (cache.count("") == 1)
352 {
353 auto iter = cache.find("");
354 ret = iter->second;
355 }
356 }
357 return ret;
358}

361/// Roll through render states given in values squentially,
362/// index starts with zero.
363const id_rstate
364traverse_states(const strings& values)
365{
366 static uint indx(0);
367 string value;
368 if (indx < values.size())
369 {
370 value = values[indx];
371 ++indx;
372 }
373 else
374 {
375 // Start over.
376 indx = 0;
377 value = values[indx];
378 }
379 return get_id_rstate(value);
380}
