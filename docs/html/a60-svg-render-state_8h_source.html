22 /**
23 Selected or Active in render area.
24 Make discrete element or layer (visible, outline, etc) if true.
25 Used as a (visibility, outline, etc.) bitmask
26 */
27 enum class select : uint
28 {
29 none = 1u << 0, ///< nothing
30 cartography = 1u << 1, ///< cartographic elements
31 vector = 1u << 2, ///< svg path, circle, rectangle, etc.
32 cloud = 1u << 3, ///< opacity color fill version of vector
33 blur = 1u << 4, ///< blur or gradient version of vector
34 echo = 1u << 5, ///< b & w outline version of vector
35 text = 1u << 6, ///< title, metadata, header
36 legend = 1u << 7, ///< text and symbol legends
37 exitnode = 1u << 8, ///< tor exit node infrastructure
38 telecom = 1u << 9, ///< telecom infrastructure
39 glyph = 1u << 10, ///< glyph
40 image = 1u << 11, ///< image
41 svg = 1u << 12, ///< svg element, perhaps nested
42 alt = 1u << 13, ///< alternate use specified in situ
43 background = 1u << 14, ///< background elements
44 raster = 1u << 15, ///< raster (pixel) elements
45 top = 1u << 16, ///< top
46 bottom = 1u << 17, ///< bottom
47 inner = 1u << 18, ///< inner
48 outer = 1u << 19, ///< outer
49 next = 1u << 20, ///< next
50 previous = 1u << 21, ///< previous
51 even = 1u << 22, ///< even
52 odd = 1u << 23, ///< odd
53 all = 1u << 24, ///< all elements and layers
54 _S_end = 1u << 25 ///< last
55 };

112 /**
113 Some high-level nobs for rendering: scale tuning.
114
115 small == defense distributed == 70k-300k
116 medium == westworld == 2-4M
117 large == stranger things == 10M+
118
119 [1-5]xsmall == reduction from least to greatest
120 1x == baseline
121 [1-5]xlarge == enlargement from least to greatest
122 */
123 enum class scale
124 {
125 r5s, r4s, r3s, r2s, r1s,
126 xxsmall, xsmall,
127 small,
128 medium, baseline,
129 large,
130 xlarge, xxlarge,
131 e1s, e2s , e3s, e4s, e5s,
132 };

167/// Settings for glyph, graphic, chart, and collection object render.
168struct render_state_base
169{
170 select visible_mode;
171
172 bool
173 is_visible(const select v) const
174 {
175 using __utype = typename std::underlying_type<select>::type;
176 __utype isv(static_cast<__utype>(visible_mode & v));
177 return isv > 0;
178 }
179};

182/// Collection derived states.
183struct collection_rstate : public render_state_base
184{
185 select outline_mode;
186 scale scale_mode;
187
188 double opacity;
189 int text_scale_max;
190
191 // Values with similar geo coordinates counted in as one meta coordinate.
192 bool weigh;
193
194 // Genderate colors from color bands (true) or use finite colors (false).
195 bool color_generated;
196
197 // Alternate output naming mode via scheme (true = info, false = map)
198 bool alt;
199
200 collection_rstate(const double o = 0.10,
201 const scale rscale = scale::medium)
202 : render_state_base(select::none), outline_mode(select::none),
203 scale_mode(rscale), opacity(o), text_scale_max(0),
204 weigh(false), color_generated(true), alt(false)
205 { }
206
207 collection_rstate(const collection_rstate&) = default;
208
209 string
210 mangle() const
211 {
212 string mangled;
213
214 bool foundp = false;
215
216 const uint vi = static_cast<uint>(visible_mode);
217 if (vi > 0)
218 {
219 mangled += "visible-mode-";
220 mangled += std::to_string(vi);
221 foundp = true;
222 }
223
224 const uint oi = static_cast<uint>(outline_mode);
225 if (oi > 0)
226 {
227 if (foundp)
228 mangled += k::hyphen;
229 else
230 foundp = true;
231 mangled += "outline-mode-";
232 mangled += std::to_string(oi);
233 }
234
235 if (opacity > 0)
236 {
237 if (foundp)
238 mangled += k::hyphen;
239 else
240 foundp = true;
241 mangled += "opacity-";
242 const double o(opacity * 100);
243 const uint oi = static_cast<uint>(o);
244 mangled += std::to_string(oi);
245 }
246
247 return mangled;
248 }
249};

282/// Named render state.
283/// Datum to take id string and tie it to visual representation.
284struct id_rstate: public render_state_base
285{
286 style styl; // Overrides render_state_base.opacity
287 string name;
288 double rotate; // Degrees to rotate to origin (CW), if any.
289 double multiple; // Scale factor, if any.
290
291 static style dstyl;
292
293 explicit
294 id_rstate(const style s = dstyl, const string f = "",
295 const double angle = 0.0, const double scale = 1.0)
296 : styl(s), name(f), rotate(angle), multiple(scale) { }
297
298 id_rstate(const id_rstate&) = default;
299 id_rstate& operator=(const id_rstate&) = default;
300};

315/// Add value to cache with base style of styl, colors klr, visibility vis.
316void
317add_to_id_rstate_cache(const string id, const style styl,
318 const select vis)
319{
320 id_rstate_umap& cache = get_id_rstate_cache();
321
322 id_rstate state(styl, id);
323 set_select(state.visible_mode, vis);
324 cache.insert(std::make_pair(id, state));
325}

328/// Given identifier/name/id, get corresponding id_rstate from cache.
329const id_rstate
330get_id_rstate(const string id)
331{
332 const id_rstate_umap& cache = get_id_rstate_cache();
333
334 id_rstate ret;
335 if (cache.count(id) == 1)
336 {
337 auto iter = cache.find(id);
338 ret = iter->second;
339 }
340 else
341 {
342 // If default string is set in the cache, use it.
343 if (cache.count("") == 1)
344 {
345 auto iter = cache.find("");
346 ret = iter->second;
347 }
348 }
349 return ret;
350}

353/// Roll through render states given in values squentially,
354/// index starts with zero.
355const id_rstate
356traverse_states(const strings& values)
357{
358 static uint indx(0);
359 string value;
360 if (indx < values.size())
361 {
362 value = values[indx];
363 ++indx;
364 }
365 else
366 {
367 // Start over.
368 indx = 0;
369 value = values[indx];
370 }
371 return get_id_rstate(value);
372}
